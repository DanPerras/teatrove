131     public void execute()
132         throws MojoExecutionException, MojoFailureException
133     {
134 
135         // Lookup build machine info
136         try
137         {
138             InetAddress address = InetAddress.getLocalHost();
139             mBuildMachine = address.getHostName();
140         }
141         catch ( UnknownHostException e )
142         {
143             //throw new MojoExecutionException( "could not get the host name", e );
144             mBuildMachine = "Unknown host";
145         }
146 
147         File outputRootDir;
148         List<String> srcRoots = project.getCompileSourceRoots();
149         for (String srcRoot : srcRoots) 
150         {
151             // validate some things
152             mSrcRoot = new File(srcRoot);
153             if ( !mSrcRoot.exists() )
154                 throw new MojoExecutionException( "srcRoot does not exist: " + mSrcRoot );
155 
156             // setup paths
157             if ( packageRoot != null && packageRoot.trim().length() > 0 )
158             {
159                 final String packageRootPath = packageRoot.replace('.', File.separatorChar);
160                 mPackageRootDir = new File( srcRoot, packageRootPath);
161                 outputRootDir = new File(outputDirectory, packageRootPath);
162             }
163             else
164             {
165                 mPackageRootDir = mSrcRoot;
166                 outputRootDir = outputDirectory;
167             }
168             getLog().debug( "packageRoot dir: " + mPackageRootDir );
169             getLog().debug( "outputDirectory dir: " + outputRootDir );
170     
171             if ( !mPackageRootDir.exists() ) {
172                 throw new MojoExecutionException( "packageRoot does not exist: " + packageRoot );
173             }
174     
175             try
176             {
177                 mSrcRootPath = mSrcRoot.getCanonicalPath();
178             }
179             catch ( IOException e )
180             {
181                 throw new MojoExecutionException( "error calling getCanonicalPath() on File object: " + srcRoot, e );
182             }
183     
184             getLog().debug( "srcRoot path: " + mSrcRootPath );
185             writePackageInfos( mPackageRootDir, outputRootDir );
186         }
187         
188         // Add the generated files as a source directory to compile against
189         project.addCompileSourceRoot(outputDirectory.getAbsolutePath());
190     }

192     /** recurse into rootDir writing PackageInfos on the way in.
193      * @param rootDir The root directory for the source
194      * @param outputDir The root directory to output to
195      * @throws org.apache.maven.plugin.MojoExecutionException
196      */
197     private void writePackageInfos( File rootDir, File outputDir )
198         throws MojoExecutionException
199     {
200         getLog().debug( "in writePackageInfos(" + rootDir + ", " + outputDir + ")" );
201 
202         try
203         {
204             if ( shouldWritePackageInfo( rootDir ) )
205             {
206                 if( !outputDir.exists() && !outputDir.mkdirs() ) {
207                     throw new MojoExecutionException( "outputDirectory was unable to be created: " + outputDir );
208                 }
209 
210                 writePackageInfo( outputDir );
211             }
212             else
213             {
214                 getLog().debug( "no files in:" + rootDir );
215             }
216 
217         }
218         catch ( Throwable e )
219         {
220             throw new MojoExecutionException( "could not write PackageInfo.java in: " + outputDir, e );
221         }
222 
223         // get any sub-directories
224         File[] subdirs = rootDir.listFiles( new FileFilter()
225         {
226             public boolean accept( File pathname )
227             {
228                 return pathname.isDirectory();
229             }
230         } );
231 
232         // recurse
233         if ( subdirs != null ) {
234             for (File subdir : subdirs) {
235                 writePackageInfos(subdir, new File(outputDir, subdir.getName()));
236             }
237         }
238     }

240     private void writePackageInfo( File dir )
241         throws IOException
242     {
243         getLog().debug( "in writePackageInfo(" + dir + ")" );
244         File outFile = new File( dir, "PackageInfo.java" );
245         PrintWriter out = new PrintWriter( new BufferedWriter( new FileWriter( outFile, false ) ) );
246 
247         try
248         {
249             // write the package statement
250             writePackageDeclaration( out, dir );
251 
252             // write the class declaration
253             writeClassDeclaration( out );
254 
255             // write the main method;
256             writeMainMethod( out );
257 
258             // write the getters;
259             writeGetters( out );
260 
261             // write the end of the class;
262             out.write( "}" + NL );
263 
264             out.flush();
265         }
266         finally
267         {
268             try
269             {
270                 out.close();
271             }
272             catch ( Throwable ignore )
273             {
274             }
275         }
276     }

278     /** Tests if a PackageInfo.java file should be written in the directory passed in.
279      *
280      *  If the File passed in is not a directory or is not writable, false is returned.
281      *  If an explicit starting package is specified to the mojo (the legacy behavior), true is returned if 
282      *     the directory passed in is a sub-package of that root package.
283      *  If an explicit starting package is NOT specified, true will only be returned if the directory passed in 
284      *     contains at least one file.
285      *
286      *  @param directory The directory to write package info to.
287      * @return returns true if a PackageInfo.java should be written.
288      */
289     private boolean shouldWritePackageInfo( File directory )
290     {
291         if ( !directory.isDirectory() )
292             return false;
293 
294         if (!directory.exists())
295             return false;
296 
297         // legacy behavior: if mPackageRootDir is a sub-dir of mSrcRootPath, an explicit package was specified
298         if(isChild(new File(mSrcRootPath), mPackageRootDir)) {
299             
300             // return true if 'directory' passed in equals() or is a child of mPackageRootDir
301             return mPackageRootDir.equals(directory) || isChild(mPackageRootDir, directory);
302             
303         } else {  // only return true if 'directory' passed in contains at least one file
304 
305             File[] files = directory.listFiles( new FileFilter()
306             {
307                 public boolean accept( File pathname )
308                 {
309                     return pathname.isFile();
310                 }
311             } );
312 
313             return files != null && files.length > 0;
314         }
315     }

317     /** @param parent The parent file.
318      * @param child The child file.
319      * @return returns true if child is anywhere in the sub-tree of directories of parent.
320      *     returns false if parent.equals(child), isChild(child, parent), or parent and child are un-related
321      */
322     private boolean isChild(File parent, File child) {
323         
324         if(parent==null || child==null) return false;
325         
326         File childsParent = child.getParentFile();
327         return parent.equals(childsParent) || isChild(parent, childsParent);
328     }

330     private void writePackageDeclaration( PrintWriter out, File packageDir )
331         throws IOException
332     {
333         if ( !mSrcRoot.equals( packageDir ) )
334         {
335 
336             String pkgPath = packageDir.getCanonicalPath();
337             String packageName = pkgPath.substring(outputDirectory.getCanonicalPath().length());
338             if ( packageName.startsWith( File.separator ) )
339                 packageName = packageName.substring( File.separator.length() );
340             packageName = packageName.replace( File.separatorChar, '.' );
341 
342             out.write( "package " + packageName + ";" + NL + NL );
343         }
344     }

426     private void writeVarAndGetter( PrintWriter out, Class<?> type, String varName, Object varValue, String getterName,
427                                    String comment )
428     {
429 
430         // write the comment
431         if ( comment != null && comment.trim().length() > 0 )
432             out.write( "/**" + NL_TAB + " * " + comment + NL_TAB + " */" + NL );
433 
434         // declare the variable and assign the value
435         out.write( TAB + "private static final " + type.getName() + " " + varName + " = "
436             + ( varValue == null ? "null" : varValue.toString() ) + ";" + NL_TAB );
437 
438         // write the getter
439         out.write( "public static " + type.getName() + " " + getterName + "() {" + NL_2TABS );
440         out.write( "return " + varName + ";" + NL_TAB );
441         out.write( "}" + NL + NL );
442     }

444     private String quote( String s )
445     {
446         // loop through the string, handling the \ characters and escaping them properly
447         if ( s == null )
448         {
449             return "\"\"";
450         }
451         StringBuffer updatedString = new StringBuffer( "\"" );
452         int i;
453         for ( i = 0; i < s.length(); i++ )
454         {
455             if ( s.charAt( i ) == '\\' )
456             {
457                 updatedString.append( "\\\\" );
458             }
459             else
460             {
461                 updatedString.append( s.charAt( i ) );
462             }
463         }
464         updatedString.append( "\"" );
465         return updatedString.toString();
466     }
