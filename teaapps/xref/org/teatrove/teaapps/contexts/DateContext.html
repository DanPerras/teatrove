62      /**
63       * Create a date object
64       * @param dateString String from which to create a date.
65       * @param inputFormat Definition of dateString.
66       * @return Date object representing date string.
67       * 
68       * <p>
69       * <strong>Time Format Syntax:</strong>
70       * <p>
71       * To specify the time format use a <em>time pattern</em> string.
72       * In this pattern, all ASCII letters are reserved as pattern letters,
73       * which are defined as the following:
74       * <blockquote>
75       * <pre>
76       * Symbol   Meaning                 Presentation        Example
77       * ------   -------                 ------------        -------
78       * G        era designator          (Text)              AD
79       * y        year                    (Number)            1996
80       * M        month in year           (Text & Number)     July & 07
81       * d        day in month            (Number)            10
82       * h        hour in am/pm (1~12)    (Number)            12
83       * H        hour in day (0~23)      (Number)            0
84       * m        minute in hour          (Number)            30
85       * s        second in minute        (Number)            55
86       * S        millisecond             (Number)            978
87       * E        day in week             (Text)              Tuesday
88       * D        day in year             (Number)            189
89       * F        day of week in month    (Number)            2 (2nd Wed in July)
90       * w        week in year            (Number)            27
91       * W        week in month           (Number)            2
92       * a        am/pm marker            (Text)              PM
93       * k        hour in day (1~24)      (Number)            24
94       * K        hour in am/pm (0~11)    (Number)            0
95       * z        time zone               (Text)              Pacific Standard Time
96       * '        escape for text         (Delimiter)
97       * ''       single quote            (Literal)           '
98       * </pre>
99       * </blockquote>
100      * The count of pattern letters determine the format.
101      * <p>
102      * <strong>(Text)</strong>: 4 or more pattern letters--use full form,
103      * &lt; 4--use short or abbreviated form if one exists.
104      * <p>
105      * <strong>(Number)</strong>: the minimum number of digits. Shorter
106      * numbers are zero-padded to this amount. Year is handled specially;
107      * that is, if the count of 'y' is 2, the Year will be truncated to 2 digits.
108      * <p>
109      * <strong>(Text & Number)</strong>: 3 or over, use text, otherwise use number.
110      * <p>
111      * Any characters in the pattern that are not in the ranges of ['a'..'z']
112      * and ['A'..'Z'] will be treated as quoted text. For instance, characters
113      * like ':', '.', ' ', '#' and '@' will appear in the resulting time text
114      * even they are not embraced within single quotes.
115      * <p>
116      * A pattern containing any invalid pattern letter will result in a thrown
117      * exception during formatting or parsing, which will result in null being
118      * returned to the template.
119      *
120      * <p>
121      * <strong>Examples:</strong>
122      * <blockquote>
123      * <pre>
124      * Format Pattern                         Result
125      * --------------                         -------
126      * "yyyy.MM.dd G 'at' hh:mm:ss z"    ->>  1996.07.10 AD at 15:08:56 PDT
127      * "EEE, MMM d, ''yy"                ->>  Wed, July 10, '96
128      * "h:mm a"                          ->>  12:08 PM
129      * "hh 'o''clock' a, zzzz"           ->>  12 o'clock PM, Pacific Daylight Time
130      * "K:mm a, z"                       ->>  0:00 PM, PST
131      * "yyyyy.MMMMM.dd GGG hh:mm aaa"    ->>  1996.July.10 AD 12:08 PM
132      * </pre>
133      * </blockquote>    
134      */
135     public Date createDate(String dateString, String inputFormat)

138         SimpleDateFormat sdf = new SimpleDateFormat(inputFormat);

176     public boolean isValidDate(int month, int day, int year, int hour, int minute, int second, int millisecond) {
177         boolean valid = true;
178         if ( month < 1 || month > 12 ) {
179             valid = false;
180         } else if ( validDay(month, day, year) == false ) {
181             valid = false;
182         } else if ( hour < 0 || hour > 23 ) {
183             valid = false;
184         } else if ( minute < 0 || minute > 59 ) {
185             valid = false;
186         } else if ( second < 0 || second > 59 ) {
187             valid = false;
188         } else if ( millisecond < 0 || millisecond > 999 ) {
189             valid = false;
190         }
191         return valid;
192     }

194     private boolean validDay(int month, int day, int year) {
195         boolean valid = true;
196         if ( month == 1 || month == 3 || month == 5 || month ==7 || month == 8 || month == 10 || month ==12 ) {
197             if ( day < 0 || day > 31 ) {
198                 valid = false;
199             }
200         } else if ( month == 4 || month == 6 || month == 9 || month == 11 ) {
201             if ( day < 0 || day > 30 ) {
202                 valid = false;
203             }
204         } else if ( month == 2 ) {
205             // if it's a leap year, they can have 29 days...
206             if ( year % 4 == 0 ) {
207                 if ( day < 0 || day > 29 ) {
208                     valid = false;
209                 }
210             } else {
211                 if ( day < 0 || day > 28 ) {
212                     valid = false;
213                 }
214             }
215         } else {
216             valid = false;
217         }
218         return valid;
219     }

244      * @param date the date from which to start
245      * @param type the type of change d for Day, M for month, y for Year

251     public Date adjustDate(Date date, String type, int delta) {
252         Date result = date;
253         if (date != null) {
254             if (DAY.equals(type)) {
255                 result = adjustDate(date, Calendar.DATE, delta);
256             } else if (MONTH.equals(type)) {
257                 result = adjustDate(date, Calendar.MONTH, delta);
258             } else if (HOUR.equals(type)) {
259                 result = adjustDate(date, Calendar.HOUR, delta);
260             } else if (YEAR.equals(type)) {
261                 result = adjustDate(date, Calendar.YEAR, delta);
262             } else if (MINUTE.equals(type)) {
263                 result = adjustDate(date, Calendar.MINUTE, delta);
264             } else if (SECOND.equals(type)) {
265                 result = adjustDate(date, Calendar.SECOND, delta);
266             }
267         }
268         return result;
269     }

272         GregorianCalendar cal = new GregorianCalendar();
273         cal.setTime(date);
274         cal.add(type, delta);
275         return cal.getTime();

301     /**
302      * Returns the age in years from today given a birthday.
303      * @param birthday the birthdate
304      * @return int the number of years that have passed 
305      *                since the birthdate. (-1 if birthday is null)
306      */
307     public int calculateAge(Date birthday) {
308         int result = -1;
309         if (birthday != null) {
310             GregorianCalendar todayCal = new GregorianCalendar();
311             todayCal.setTime(new Date());
312             int dayOfYear = todayCal.get(Calendar.DAY_OF_YEAR);
313             int year = todayCal.get(Calendar.YEAR);
314             GregorianCalendar birthdayCal = new GregorianCalendar();
315             birthdayCal.setTime(birthday);
316             int birthDayOfYear = birthdayCal.get(Calendar.DAY_OF_YEAR);
317             int birthYear = birthdayCal.get(Calendar.YEAR);
318             birthDayOfYear = processLeapYear(todayCal, birthdayCal, birthDayOfYear);
319             result = year - birthYear;
320             if (dayOfYear < birthDayOfYear) {
321                 result--;
322             }
323         }
324         return result;
325     }
326     
327     private int processLeapYear(GregorianCalendar today, GregorianCalendar birthdayCal, int value) {    
328         int result = value;
329         boolean isLeapYear = today.isLeapYear(today.get(Calendar.YEAR));
330         boolean isFebruary = (birthdayCal.get(Calendar.MONTH) == Calendar.FEBRUARY);
331         boolean isLeapDay = (birthdayCal.get(Calendar.DAY_OF_MONTH) == 29);
332         if (isFebruary && (!isLeapYear) && isLeapDay) {
333             result--;
334         }
335         return result;
336     }
