39      /**
40       * Method to set attributes of an object.  This method may be 
41       * called to set the values of an object by passing in the object, 
42       * the attribute name to set, and the attribute value.  To get a 
43       * listing of attribute values that can be set for a given object, 
44       * see the createObject method for the object.
45       * <P>
46       * @param object The object to set an attribute value for.
47       * @param attributeName The name of the attribute to set.
48       * @param attributeValue The value of the attribute to set.
49       */
50      public void set(Object object, String attributeName, Object attributeValue)
51          throws IllegalAccessException, InvocationTargetException, BeanContextException
52      {
53          Class<?> objectClass = object.getClass();
54          String methodName = "set" + attributeName.substring(0,1).toUpperCase() + attributeName.substring(1);        
55          String methodKey = objectClass.getName() + "." + methodName;
56          MethodTypePair pair = (MethodTypePair) mMethodTypePairMap.get(methodKey);
57          if (pair == null) {
58              pair = getMethodTypePair(objectClass, methodName, attributeValue);
59              mMethodTypePairMap.put(methodKey, pair);
60          }
61          if (pair != null) {        
62              Object[] params = {convertObjectType(attributeValue, pair.type)};
63              pair.method.invoke(object, params);            
64          } else {
65              if (objectClass != null) {
66                  throw new BeanContextException("A method " + objectClass.getName() + ".");
67              }
68              if (methodName != null && attributeValue != null) {
69                  throw new BeanContextException(methodName + "(" + attributeValue.getClass() + ") could not be found.");
70              }
71          }
72      }

74      private static MethodTypePair getMethodTypePair(Class<?> objectClass, String methodName, Object attributeValue) {
75          MethodTypePair result = null;
76          Method method;
77          Class<?> valueClass = null;
78          if (attributeValue != null) {
79              valueClass = attributeValue.getClass();
80              try {
81                  Class<?>[] param = new Class[] {toPrimitiveType(valueClass)};
82                  method = objectClass.getMethod(methodName, param);
83                  result = new MethodTypePair(method, valueClass);
84              } catch (NoSuchMethodException e) {
85                  Class<?>[] compatibleClasses = getCompatibleClasses(valueClass);
86                  if (compatibleClasses != null) {
87                      for (int i=0; i < compatibleClasses.length; i++) {
88                          try {
89                              Class<?>[] param = new Class[] {toPrimitiveType(compatibleClasses[i])};
90                              method = objectClass.getMethod(methodName, param);
91                              result = new MethodTypePair(method, compatibleClasses[i]);
92                              break;
93                          } catch (NoSuchMethodException e2) {
94                          }
95                      }
96                  }
97              }
98          } else {
99              Method[] methods = objectClass.getMethods();
100             for (int i=0; i < methods.length; i++) {
101                 String name = methods[i].getName();
102                 Class<?>[] clazzes = methods[i].getParameterTypes();
103                 if (name.equals(methodName) && 
104                     clazzes.length == 1 &&
105                     !clazzes[0].isPrimitive()) {
106                     result = new MethodTypePair(methods[i], clazzes[0]);
107                     break;
108                 }
109             }
110         }
111         return result;
112     }

114     private static Class<?> toPrimitiveType(Class<?> clazz) {
115         Class<?> result = clazz;
116         if (clazz != null) {
117             TypeDesc typeDesc = TypeDesc.forClass(clazz);
118             typeDesc = typeDesc.toPrimitiveType();
119             if (typeDesc != null) {
120                 result = typeDesc.toClass();
121             }
122         }
123         return result;
124     }

126     private static Class<?>[] getCompatibleClasses(Class<?> type) {
127         Class<?>[] result = null;
128         if (type == Integer.class) {
129             result = new Class[] {Long.class, Short.class, Byte.class};
130         } else if (type == Long.class) {
131             result = new Class[] {Integer.class, Short.class, Byte.class};
132         } else if (type == Short.class) {
133             result = new Class[] {Long.class, Integer.class, Byte.class};
134         } else if (type == Byte.class) {
135             result = new Class[] {Long.class, Integer.class, Short.class};
136         } else if (type == Double.class) {
137             result = new Class[] {Float.class};
138         } else if (type == Float.class) {
139             result = new Class[] {Double.class};
140         }
141         return result;
142     }

144     private static Object convertObjectType(Object value, Class<?> typeToCastInto) {
145         Object result = value;
146         if (value != null) {
147             Class<?> valueClass = value.getClass();
148             if (!(valueClass == typeToCastInto)) {
149                 Number numberValue = (Number) value;
150                 if (valueClass == Integer.class) {
151                     if (typeToCastInto == Long.class) {
152                         result = new Long(numberValue.longValue());
153                     } else if (typeToCastInto == Short.class) {
154                         result = new Short(numberValue.shortValue());
155                     } else if (typeToCastInto == Byte.class) {
156                         result = new Byte(numberValue.byteValue());
157                     }
158                 } else if (valueClass == Double.class) {
159                     if (typeToCastInto == Float.class) {
160                         result = new Float(numberValue.floatValue());
161                     }
162                 }
163             }
164         }
165         return result;
166     }

168     public static class MethodTypePair {
169         Method method;
170         Class<?> type;
171 
172         public MethodTypePair(Method method, Class<?> type) {
173             this.method = method;
174             this.type = type;
175         }

178     public class BeanMap {
179         private Object mBean;
180         private BeanPropertyAccessor mAccessor;
181         
182         public BeanMap(Object bean, BeanPropertyAccessor accessor) {
183             mBean = bean;
184             mAccessor = accessor;
185         }
186         
187         public Object get(String property) {
188             return mAccessor.getPropertyValue(mBean, property);
189         }
190     }
