205     /**
206      * Opens a connection, passes on the current request settings, and returns
207      * the server's response. The optional PostData parameter is used to
208      * supply post data to the server. The Content-Length header specifies
209      * how much data will be read from the PostData InputStream. If it is not
210      * specified, data will be read from the InputStream until EOF is reached.
211      *
212      * @param postData additional data to supply to the server, if request
213      * method is POST
214      */
215     public Response getResponse(PostData postData)
216         throws ConnectException, SocketException
217     {
218         CheckedSocket socket = mFactory.getSocket(mSession);
219 
220         try {
221             CharToByteBuffer request = new FastCharToByteBuffer
222                 (new DefaultByteBuffer(), "8859_1");
223             request = new InternedCharToByteBuffer(request);
224 
225             request.append(mMethod);
226             request.append(' ');
227             request.append(mURI);
228             request.append(' ');
229             request.append(mProtocol);
230             request.append("\r\n");
231             if (mHeaders != null) {
232                 mHeaders.appendTo(request);
233             }
234             request.append("\r\n");
235 
236             Response response;
237             try {
238                 response = sendRequest(socket, request, postData);
239             }
240             catch (InterruptedIOException e) {
241                 // If timed out, throw exception rather than risk double
242                 // posting.
243                 throw e;
244             }
245             catch (IOException e) {
246                 response = null;
247             }
248 
249             if (response == null) {
250                 // Try again with new connection. Persistent connection may
251                 // have timed out and been closed by server.
252                 try {
253                     socket.close();
254                 }
255                 catch (IOException e) {
256                 }
257 
258                 socket = mFactory.createSocket(mSession);
259 
260                 response = sendRequest(socket, request, postData);
261                 if (response == null) {
262                     throw new ConnectException("No response from server " + socket.getInetAddress() + ":" +
263                             socket.getPort());
264                 }
265             }
266 
267             return response;
268         }
269         catch (SocketException e) {
270             throw e;
271         }
272         catch (InterruptedIOException e) {
273             throw new ConnectException("Read timeout expired: " +
274                                        mReadTimeout + ", " + e);
275         }
276         catch (IOException e) {
277             throw new SocketException(e.toString());
278         }
279     }

281     private Response sendRequest(CheckedSocket socket,
282                                  ByteBuffer request,
283                                  PostData postData)
284         throws SocketException, IOException
285     {
286         socket.setSoTimeout(mReadTimeout);
287 
288         OutputStream out =
289             new FastBufferedOutputStream(socket.getOutputStream());
290         request.writeTo(out);
291         if (postData != null) {
292             writePostData(out, postData);
293         }
294         out.flush();
295         InputStream in = new FastBufferedInputStream(socket.getInputStream());
296 
297         char[] buf = new char[100];
298         String line = HttpUtils.readLine(in, buf);
299 
300         if (line == null) {
301             return null;
302         }
303 
304         return new Response(socket, mMethod, in, buf, line);
305     }

307     private void writePostData(OutputStream out, PostData postData)
308         throws IOException
309     {
310         InputStream in = postData.getInputStream();
311 
312         int contentLength = -1;
313         if (mHeaders != null) {
314             Integer i = mHeaders.getInteger("Content-Length");
315             if (i != null) {
316                 contentLength = i.intValue();
317             }
318         }
319 
320         byte[] buf;
321         if (contentLength < 0 || contentLength > 4000) {
322             buf = new byte[4000];
323         }
324         else {
325             buf = new byte[contentLength];
326         }
327 
328         try {
329             int len;
330             if (contentLength < 0) {
331                 while ((len = in.read(buf)) > 0) {
332                     out.write(buf, 0, len);
333                 }
334             }
335             else {
336                 while (contentLength > 0) {
337                     len = buf.length;
338                     if (contentLength < len) {
339                         len = contentLength;
340                     }
341                     if ((len = in.read(buf, 0, len)) <= 0) {
342                         break;
343                     }
344                     out.write(buf, 0, len);
345                     contentLength -= len;
346                 }
347             }
348         }
349         finally {
350             in.close();
351         }
352     }

366     public class Response {
367         private final CheckedSocket mSocket;
368         private final String mRequestMethod;

419          */

421             return mStatusCode == 100;
422         }
423 
424         /**
425          * If status code is 100 (continue), then call nextResponse to read the
426          * next response from the server. Returns false if current response
427          * status code is not 100. Returns true if next response was read,
428          * updating this response's status code and headers.
429          *
430          * If the getInputStream method is called before calling nextResponse,
431          * then the response is automatically moved to the next non-100
432          * response.
433          */
434         public boolean nextResponse() throws IOException {
435             if (mStatusCode != 100) {
436                 return false;
437             }
438             char[] buf = new char[100];
439             String line = HttpUtils.readLine(mRawIn, buf);
440             if (line == null) {
441                 throw new ProtocolException("No next response");
442             }
443             nextResponse(buf, line);
444             return true;
445         }
446 
447         /**
448          * Returns an InputStream supplying the body of the response. When all
449          * of the response body has been read, the connection is either closed
450          * or recycled, depending on if all the criteria is met for supporting
451          * persistent connections. Further reads on the InputStream will
452          * return EOF.
453          */
454         public InputStream getInputStream() throws IOException {
455             if (mIn != null) {
456                 return mIn;
457             }
458 
459             while (nextResponse()) {
460             }
461 
462             // Used for controlling persistent connections.
463             int contentLength;
464             if ("Keep-Alive".equalsIgnoreCase
465                 (mHeaders.getString("Connection"))) {
466 
467                 if ("HEAD".equals(mRequestMethod)) {
468                     contentLength = 0;
469                 }
470                 else {
471                     Integer i = mHeaders.getInteger("Content-Length");
472                     if (i != null) {
473                         contentLength = i.intValue();
474                     }
475                     else {
476                         contentLength = -1;
477                     }
478                 }
479             }
480             else {
481                 contentLength = -1;
482             }
483 
484             return mIn = new ResponseInput(mSocket, mRawIn, contentLength);
485         }
486 
487         private void nextResponse(char[] buf, String line) throws IOException {
488             int statusCode = -1;
489             String statusMessage = "";
490 
491             int space = line.indexOf(' ');
492             if (space > 0) {
493                 int nextSpace = line.indexOf(' ', space + 1);
494                 String sub;
495                 if (nextSpace < 0) {
496                     sub = line.substring(space + 1);
497                 }
498                 else {
499                     sub = line.substring(space + 1, nextSpace);
500                     statusMessage = line.substring(nextSpace + 1);
501                 }
502                 try {
503                     statusCode = Integer.parseInt(sub);
504                 }
505                 catch (NumberFormatException e) {
506                 }
507             }
508 
509             if (statusCode < 0) {
510                 throw new ProtocolException("Invalid HTTP response: " + line);
511             }
512 
513             mStatusCode = statusCode;
514             mStatusMessage = statusMessage;
515             mHeaders = new HttpHeaderMap();
516             mHeaders.readFrom(mRawIn, buf);
517         }
518     }

520     private class ResponseInput extends InputStream {
521         private CheckedSocket mSocket;
522         private InputStream mIn;
523         private int mContentLength;
524 
525         /**
526          * @param contentLength Used for supporting persistent connections. If
527          * negative, then close connection when EOF is read.
528          */
529         public ResponseInput(CheckedSocket socket,
530                              InputStream in, int contentLength)
531             throws IOException
532         {
533             mSocket = socket;
534             mIn = in;
535             if ((mContentLength = contentLength) == 0) {
536                 recycle();
537             }
538         }
539 
540         public int read() throws IOException {
541             if (mContentLength == 0) {
542                 return -1;
543             }
544 
545             int b = mIn.read();
546 
547             if (b < 0) {
548                 close();
549             }
550             else if (mContentLength > 0) {
551                 if (--mContentLength == 0) {
552                     recycle();
553                 }
554             }
555 
556             return b;
557         }
558 
559         public int read(byte[] b) throws IOException {
560             return read(b, 0, b.length);
561         }
562 
563         public int read(byte[] b, int off, int len) throws IOException {
564             if (mContentLength == 0) {
565                 return -1;
566             }
567 
568             if (mContentLength < 0) {
569                 len = mIn.read(b, off, len);
570                 if (len < 0) {
571                     close();
572                 }
573                 else if (len == 0) {
574                     close();
575                     len = -1;
576                 }
577                 return len;
578             }
579 
580             if (len > mContentLength) {
581                 len = mContentLength;
582             }
583             else if (len == 0) {
584                 return 0;
585             }
586 
587             len = mIn.read(b, off, len);
588 
589             if (len < 0) {
590                 close();
591             }
592             else if (len == 0) {
593                 close();
594                 len = -1;
595             }
596             else {
597                 if ((mContentLength -= len) == 0) {
598                     recycle();
599                 }
600             }
601 
602             return len;
603         }
604 
605         public long skip(long n) throws IOException {
606             if (mContentLength == 0) {
607                 return 0;
608             }
609 
610             if (mContentLength < 0) {
611                 return mIn.skip(n);
612             }
613 
614             if (n > mContentLength) {
615                 n = mContentLength;
616             }
617             else if (n == 0) {
618                 return 0;
619             }
620 
621             n = mIn.skip(n);
622 
623             if ((mContentLength -= n) == 0) {
624                 recycle();
625             }
626 
627             return n;
628         }
629 
630         public int available() throws IOException {
631             if (mContentLength == 0) {
632                 return 0;
633             }
634             else {
635                 try {
636                     return mIn.available();
637                 }
638                 catch (SocketException se) {
639                     return 0;
640                 }
641             }
642         }
643 
644         public void close() throws IOException {
645             if (mSocket != null) {
646                 mContentLength = 0;
647                 mSocket = null;
648                 mIn.close();
649             }
650         }
651 
652         private void recycle() throws IOException {
653             if (mSocket != null) {
654                 if (mContentLength == 0) {
655                     CheckedSocket s = mSocket;
656                     mSocket = null;
657                     mFactory.recycleSocket(s);
658                 }
659                 else {
660                     mSocket = null;
661                     mIn.close();
662                 }
663             }
664         }
665     }
