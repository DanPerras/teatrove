160     // Returns false if no more results should be added.
161     protected static boolean addMatchResult(int limit,
162                                             List results,
163                                             String pattern,
164                                             Object value,
165                                             int[] positions,
166                                             int len)
167     {
168         int size = results.size();
169         if (size < limit) {
170             if (positions == null || len == 0) {
171                 positions = NO_POSITIONS;
172             }
173             else {
174                 int[] original = positions;
175                 positions = new int[len];
176                 for (int i=0; i<len; i++) {
177                     positions[i] = original[i];
178                 }
179             }
180             results.add(new Result(pattern, value, positions));
181             return size + 1 < limit;
182         }
183         else {
184             return false;
185         }
186     }

246         Maker(Map patternMap) {
247             String[] keys =
248                 (String[])patternMap.keySet().toArray(new String[0]);
249 
250             for (int i=0; i<keys.length; i++) {
251                 String key = keys[i];
252                 // Ensure terminating patterns end in the special
253                 // terminator char.
254                 if (!key.endsWith("*")) {
255                     keys[i] = key.concat("\uffff");
256                 }
257             }
258 
259             // Sort the keys in a special order that ensures correct
260             // "closest match" semantics.
261             Arrays.sort(keys, new PatternComparator());
262 
263             mMappedValues = new Object[keys.length];
264             for (int i=0; i<keys.length; i++) {
265                 String key = keys[i];
266                 if (key.endsWith("\uffff")) {
267                     key = key.substring(0, key.length() - 1);
268                 }
269                 mMappedValues[i] = patternMap.get(key);
270             }
271             
272             // Build tree structure for managing pattern matching.
273             mPatternRoot = new PatternNode();
274             for (int i=0; i<keys.length; i++) {
275                 String key = keys[i];
276                 mPatternRoot.buildPathTo(key, i);
277             }
278 
279             mMaxWildPerKey = mPatternRoot.getMaxWildcardCount();
280 
281             mKey = new MultiKey(keys);
282         }

351         private void generateBranches(PatternNode node, int depth,
352                                       int posIndex,
353                                       LocalVariable tempChar) {
354             int c = node.mChar;
355             List subNodes = node.mSubNodes;
356 
357             mBuilder.mapLineNumber(++mReferenceLine);
358 
359             if (c == '*') {
360                 LocalVariable savedIndex;
361                 
362                 if (mTempLocals.isEmpty()) {
363                     savedIndex =
364                         mBuilder.createLocalVariable("temp", mIntType);
365                 }
366                 else {
367                     savedIndex = (LocalVariable)mTempLocals.pop();
368                 }
369                 
370                 mBuilder.loadLocal(mIndexLocal);
371                 mBuilder.storeLocal(savedIndex);
372                 
373                 // Save position of wildcard start.
374                 mBuilder.loadLocal(mPositionsLocal);
375                 mBuilder.loadConstant(posIndex);
376                 mBuilder.loadLocal(mIndexLocal);
377                 if (depth > 0) {
378                     mBuilder.loadConstant(depth);
379                     mBuilder.math(Opcode.IADD);
380                 }
381                 mBuilder.storeToArray(TypeDesc.INT);
382                 
383                 if (subNodes == null) {
384                     generateWildcard(null, depth, posIndex + 2);
385                 }
386                 else {
387                     int size = subNodes.size();
388                     for (int i=0; i<size; i++) {
389                         generateWildcard((PatternNode)subNodes.get(i),
390                                          depth, posIndex + 2);
391                         mBuilder.loadLocal(savedIndex);
392                         mBuilder.storeLocal(mIndexLocal);
393                     }
394                 }
395                 
396                 mTempLocals.push(savedIndex);
397 
398                 if (node.mPattern != null) {
399                     generateAddMatchResult(node);
400                 }
401 
402                 return;
403             }
404 
405             Label noMatch = mBuilder.createLabel();
406 
407             if (c >= 0) {
408                 if (tempChar != null) {
409                     mBuilder.loadLocal(tempChar);
410                     mTempLocals.push(tempChar);
411                 }
412                 else {
413                     mBuilder.loadLocal(mLookupLocal);
414                     mBuilder.loadLocal(mIndexLocal);
415                     if (depth > 0) {
416                         mBuilder.loadConstant(depth);
417                         mBuilder.math(Opcode.IADD);
418                     }
419                     mBuilder.loadFromArray(TypeDesc.CHAR);
420                 }
421                 
422                 mBuilder.loadConstant((char)c);
423                 mBuilder.ifComparisonBranch(noMatch, "!=");
424             }
425             
426             if (subNodes != null) {
427                 int size = subNodes.size();
428                 for (int i=0; i<size; i++) {
429                     generateBranches
430                         ((PatternNode)subNodes.get(i), depth + 1, posIndex);
431                 }
432             }
433             
434             if (node.mPattern != null) {
435                 // Matched pattern; save results.
436                 generateAddMatchResult(node);
437             }
438 
439             noMatch.setLocation();
440         }

442         private void generateWildcard(PatternNode node, int depth,
443                                       int posIndex) {
444             Label loopStart = mBuilder.createLabel().setLocation();
445             Label loopEnd = mBuilder.createLabel();
446             Label loopContinue = mBuilder.createLabel();
447 
448             // Save position of wildcard end.
449             mBuilder.loadLocal(mPositionsLocal);
450             mBuilder.loadConstant(posIndex - 1);
451             mBuilder.loadLocal(mIndexLocal);
452             if (depth > 0) {
453                 mBuilder.loadConstant(depth);
454                 mBuilder.math(Opcode.IADD);
455             }
456             mBuilder.storeToArray(TypeDesc.INT);
457 
458             mBuilder.loadLocal(mLookupLocal);
459             mBuilder.loadLocal(mIndexLocal);
460             if (depth > 0) {
461                 mBuilder.loadConstant(depth);
462                 mBuilder.math(Opcode.IADD);
463             }
464             mBuilder.loadFromArray(TypeDesc.CHAR);
465 
466             if (node == null) {
467                 mBuilder.loadConstant('\uffff');
468                 mBuilder.ifComparisonBranch(loopEnd, "==");
469             }
470             else {
471                 LocalVariable tempChar;
472                 if (mTempLocals.isEmpty()) {
473                     tempChar =
474                         mBuilder.createLocalVariable("temp", mIntType);
475                 }
476                 else {
477                     tempChar = (LocalVariable)mTempLocals.pop();
478                 }
479                 mBuilder.storeLocal(tempChar);
480                 mBuilder.loadLocal(tempChar);
481 
482                 mBuilder.loadConstant('\uffff');
483                 mBuilder.ifComparisonBranch(loopEnd, "==");
484 
485                 generateBranches(node, depth, posIndex, tempChar);
486             }
487 
488             loopContinue.setLocation();
489             mBuilder.integerIncrement(mIndexLocal, 1);
490             mBuilder.branch(loopStart);
491             loopEnd.setLocation();
492         }

494         private void generateAddMatchResult(PatternNode node) {
495             mBuilder.mapLineNumber(++mReferenceLine);
496 
497             mBuilder.loadLocal(mLimitLocal);
498             mBuilder.loadLocal(mResultsLocal);
499             mBuilder.loadConstant(node.mPattern);
500             mBuilder.loadThis();
501             mBuilder.loadField("mValues", TypeDesc.OBJECT.toArrayType());
502             mBuilder.loadConstant(node.mOrder);
503             mBuilder.loadFromArray(TypeDesc.OBJECT);
504             mBuilder.loadLocal(mPositionsLocal);
505             mBuilder.loadConstant(node.getWildcardCount() * 2);
506 
507             TypeDesc[] params = {
508                 mIntType,
509                 mListType,
510                 mStringType,
511                 mObjectType,
512                 mIntArrayType,
513                 mIntType
514             };
515             
516             mBuilder.invokeStatic(PatternMatcher.class.getName(),
517                                   "addMatchResult", mBooleanType, params);
518             mBuilder.ifZeroComparisonBranch(mReturnLabel, "==");
519         }

584         private void buildPathTo(String pattern, int order, int index) {
585             if (index >= pattern.length()) {
586                 if (pattern.endsWith("\uffff")) {
587                     // Trim off the '\uffff'.
588                     pattern = pattern.substring(0, pattern.length() - 1);
589                 }
590                 mPattern = pattern;
591                 mOrder = order;
592                 return;
593             }
594 
595             char c = pattern.charAt(index);
596 
597             if (mSubNodes == null) {
598                 mSubNodes = new ArrayList(10);
599             }
600 
601             int size = mSubNodes.size();
602             for (int i=0; i<size; i++) {
603                 PatternNode node = (PatternNode)mSubNodes.get(i);
604                 if (node.mChar == c) {
605                     node.buildPathTo(pattern, order, index + 1);
606                     return;
607                 }
608             }
609 
610             PatternNode node = new PatternNode(c);
611             mSubNodes.add(node);
612             node.buildPathTo(pattern, order, index + 1);
613 
614             return;
615         }

636     private static class PatternComparator implements Comparator {
637         public int compare(Object a, Object b) {
638             String sa = (String)a;
639             String sb = (String)b;
640             
641             int alen = sa.length();
642             int blen = sb.length();
643             int mlen = Math.min(alen, blen);
644             
645             for (int i=0; i<mlen; i++) {
646                 char ca = sa.charAt(i);
647                 char cb = sb.charAt(i);
648                 if (ca == '*') {
649                     if (cb != '*') {
650                         // Wildcard sorted high.
651                         return 1;
652                     }
653                 }
654                 else if (cb == '*') {
655                     // Wildcard sorted high.
656                     return -1;
657                 }
658                 else if (ca < cb) {
659                     return -1;
660                 }
661                 else if (ca > cb) {
662                     return 1;
663                 }
664             }
665             
666             // The shorter string is sorted high.
667             if (alen < blen) {
668                 return 1;
669             }
670             else if (alen > blen) {
671                 return -1;
672             }
673             
674             return 0;
675         }
