17  package org.teatrove.trove.util;
18  
19  /**
20   * A factory for creating Integer objects. Under low load, creating new Integer
21   * objects is quicker than using this factory. Under heavly load, this factory
22   * can offer performance benefits because the garbage collector runs less
23   * frequently.
24   *
25   * @author Brian S O'Neill
26   */
27  public final class IntegerFactory {
28      private static final Integer[] CACHE;
29      private static final int CACHE_MASK;
30  
31      static {
32          Integer i = Integer.getInteger
33              ("org.teatrove.trove.util.IntegerFactory.capacity");
34          int cacheSize = (i == null) ? 1024 : i.intValue();
35          // Ensure cache size is even power of 2.
36          cacheSize--;
37          int shift = 0;
38          while (cacheSize > 0) {
39              shift++;
40              cacheSize >>= 1;
41          }
42          cacheSize = 1 << shift;
43          CACHE = new Integer[cacheSize];
44          CACHE_MASK = cacheSize - 1;
45      }
46  
47      // Although accessed by multiple threads, this method doesn't need to be
48      // synchronized.
49      public static Integer toInteger(int value) {
50          Integer[] cache = CACHE;
51          int index = value & CACHE_MASK;
52          Integer v = cache[index];
53          if (v == null || v.intValue() != value) {
54              v = new Integer(value);
55              cache[index] = v;
56          }
57          return v;
58      }
59  
60      private IntegerFactory() {
61      }
62  }
