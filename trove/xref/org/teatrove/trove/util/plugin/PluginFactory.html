34  /**
35   * This class is responsible for creating plugins based from a
36   * configuration block.
37   *
38   * @author Scott Jappinen, Travis Greer
39   */
40  public class PluginFactory {
41  
42      private static final String cClassKey = "class";
43      private static final String cInitKey = "init";
44      private static final String cPluginsKey = "plugins";
45      private static final String cInjectedFileKey = "propertyInjector";
46  	private static final String cExternalPropKey = "externalProperty";
47  	private static final String cExternalNameKey = "externalPluginName";
48  
49  
50      public static final Plugin createPlugin(String name, PluginFactoryConfig config)
51          throws PluginFactoryException
52      {
53          Plugin result;
54          String className = config.getProperties().getString(cClassKey);
55          PropertyMap props = config.getProperties().subMap(cInitKey);
56  
57  
58          if (config.getProperties().containsKey(cInjectedFileKey)) {
59          	try {
60          		if (props == null) {
61          			props = new PropertyMap();
62          		}
63  
64          		Map injectedProps = new PropertyMap();
65  	        	PropertyParser parser = new PropertyParser(props);
66  				parser.parse(new FileReader((String)config.getProperties().get(cInjectedFileKey)));
67  				props.putAll(injectedProps);
68          	} catch (IOException e) {
69          		new PluginFactoryException(e);
70  			} catch (Exception e) {
71          		new PluginFactoryException(e);
72  			}
73          }
74  
75          //This block is used to retrieve plugins from an external file
76  		if (config.getProperties().containsKey(cExternalPropKey)) {
77  			try {
78  				if (props == null) {
79  					props = new PropertyMap();
80  				}
81  
82  				PropertyMap injectedProps = new PropertyMap();
83  				String externalPlugin = config.getProperties().getString(cExternalNameKey);
84  				Document pluginsDoc = null;
85  				pluginsDoc = XMLMapFactory.createDocument(new FileReader((String)config.getProperties().get(cExternalPropKey)));
86  				
87  				if(pluginsDoc != null) {
88  					Element rootElement = pluginsDoc.getRootElement();
89  
90  					if(rootElement != null) {
91  						Element child = null;
92  						//Plugin name in the services and external config may be different
93  						if(externalPlugin != null) {
94  							config.getLog().debug(name + " is loading the external plugin " + externalPlugin);
95  							child = rootElement.getChild(externalPlugin);
96  						}
97  						else {
98  							child = rootElement.getChild(name);
99  						}
100 						if(child != null) {
101 							injectedProps = XMLMapFactory.getPropertyMapFromElement(child);
102 
103 							//if the class name wasn't specified, get it from the external config
104 							if(className == null) {
105 								className = child.getChildText(cClassKey);
106 							}
107 						}
108 					}
109 				}
110 				if(injectedProps != null) {
111 					if(externalPlugin != null) {
112 						props.putAll(injectedProps.subMap(externalPlugin).subMap(cInitKey));
113 					}
114 					else {
115 						props.putAll(injectedProps.subMap(name).subMap(cInitKey));
116 					}
117 				}
118 			} catch (Exception e) {
119 				throw new PluginFactoryException("Error loading external properties for " + name, e);
120 			}
121 		}//END external file load block
122 		
123         try {
124             Class clazz = Class.forName(className);
125             result = (Plugin) clazz.newInstance();
126             PluginConfig pluginConfig = new PluginConfigSupport
127                 (props, config.getLog(), config.getPluginContext(), name);
128             result.init(pluginConfig);
129         } catch (PluginException e) {
130             throw new PluginFactoryException(e);
131         } catch (ClassNotFoundException e) {
132             throw new PluginFactoryException(e);
133         } catch (InstantiationException e) {
134             throw new PluginFactoryException(e);
135         } catch (IllegalAccessException e) {
136             throw new PluginFactoryException(e);
137         }
138         return result;
139     }

141     public static final Plugin[] createPlugins(PluginFactoryConfig config)
142         throws PluginFactoryException
143     {
144         Plugin[] result;
145         Map plugins;
146         PropertyMap properties = config.getProperties().subMap(cPluginsKey);
147         if (properties.containsKey("properties.file")) {
148             String defaultsFileName = (String)properties.remove("properties.file");
149             loadDefaults(defaultsFileName, properties, config.getLog());
150         }
151 
152         Set keySet = properties.subMapKeySet();
153         plugins = new HashMap(keySet.size());
154         Iterator iterator = keySet.iterator();
155         for (int i=0; iterator.hasNext(); i++) {
156             String name = (String) iterator.next();
157             PropertyMap initProps = properties.subMap(name);
158             PluginFactoryConfig conf = new PluginFactoryConfigSupport
159                 (initProps, config.getLog(), config.getPluginContext());
160             try {
161 	            plugins.put(name, createPlugin(name, conf));
162 			} catch (Exception e) {
163 				// catch everything so we can continue loading even if one plugin is broken
164 				config.getLog().error("Error loading plugin: " + name);
165 				config.getLog().error(e);
166 			}
167         }
168         Map registeredPlugins = config.getPluginContext().getPlugins();
169         keySet = registeredPlugins.keySet();
170         iterator = keySet.iterator();
171         while(iterator.hasNext()) {
172             String name = (String) iterator.next();
173             if (!plugins.containsKey(name)) {
174                 plugins.put(name, registeredPlugins.get(name));
175             }
176         }
177         result = new Plugin[plugins.size()];
178         result = (Plugin[])plugins.values().toArray(result);
179         return result;
180     }
181 
182     private static void loadDefaults(String filename, PropertyMap props, Log log) {
183         log.info("loading Plugin defaults");
184         try {
185             Map defaults = new PropertyMap();
186             FileReader fr = new FileReader(filename);
187             PropertyParser pp = new PropertyParser(defaults);
188             pp.parse(fr);
189             log.info("loaded " + defaults.size() + " default keys");
190             props.putDefaults(defaults);
191         }
192         catch (IOException ioe) {
193             log.error(ioe);
194         }
195     }
196 }
