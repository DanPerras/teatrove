53  /**
54   * Merges several classes together, producing a new class that has all of the
55   * methods of the combined classes. All methods in the combined class delegate
56   * to instances of the source classes. If multiple classes implement the same
57   * method, the first one provided is used. The merged class implements all of
58   * the interfaces provided by the source classes or interfaces.
59   *
60   * <p>This class performs a function almost the same as the Proxy class
61   * introduced in JDK1.3. It differs in that it supports classes as well as
62   * interfaces as input, and it binds to wrapped objects without using
63   * runtime reflection. It is less flexible than Proxy in that there isn't way
64   * to customize the method delegation, and so it isn't suitable for creating
65   * dynamically generated interface implementations.
66   *
67   * @author Brian S O'Neill, Nick Hagan
68   */
69  public class MergedClass {
70      // Maps ClassInjectors to Maps that map to merged class names.
71      // Map<ClassInjector, Map<MultiKey, String>>
72      // The MultiKey is composed of class names and method prefixes. By storing
73      // class names into the maps instead of classes, the classes may be
74      // reclaimed by the garbage collector.
75      private static Map cMergedMap;
76  
77      public static final int OBSERVER_DISABLED = 0;  // Invocation events disabled
78      public static final int OBSERVER_ENABLED = 1;   // Invocation events enabled
79      public static final int OBSERVER_ACTIVE = 2;    // Invocation events enabled and active.
80      public static final int OBSERVER_EXTERNAL = 4;    // Invocation events enabled but triggered externally.

168     /**
169      * Returns the constructor for a class that merges all of the given source
170      * classes. The constructor's parameter types match the source classes.
171      * An IllegalArgumentException is thrown if any of the given conditions
172      * are not satisfied:
173      *
174      * <ul>
175      * <li>None of the given classes can represent a primitive type
176      * <li>A source class can only be provided once, unless paired with a
177      * unique prefix.
178      * <li>Any non-public classes must be in the same common package
179      * <li>Duplicate methods cannot have conflicting return types, unless a
180      * prefix is provided.
181      * <li>The given classes must be loadable from the given injector
182      * <li>At most 254 classes may be merged
183      * </ul>
184      *
185      * To help resolve ambiguities, a method prefix can be specified for each
186      * passed in class. For each prefixed method, the non-prefixed method is
187      * also generated, unless that method conflicts with the return type of
188      * another method. If any passed in classes are or have interfaces, then
189      * those interfaces will be implemented only if there are no conflicts.
190      * An optional implementation of the MethodInvocationEventObserver
191      * interface may be supplied.
192      *
193      * <p>The array of prefixes may be null, have null elements or
194      * be shorter than the array of classes. A prefix of "" is treated as null.
195      * <p>
196      * Note: Because a constructor is limited to 254 parameters, if more than
197      * 254 classes are to be merged together, call {@link #getConstructor2}.
198      *
199      * @param injector ClassInjector that will receive class definition
200      * @param classes Source classes used to derive merged class
201      * @param prefixes Optional prefixes to apply to methods of each generated
202      * class to eliminate duplicate method names
203      * @param observerMode int Invocation event handling modes.
204      */
205     public static Constructor getConstructor(ClassInjector injector,
206                                              Class[] classes,
207                                              String[] prefixes,
208                                              int observerMode)
209         throws IllegalArgumentException
210     {
211         if (classes.length > 254) {
212             throw new IllegalArgumentException
213                 ("More than 254 merged classes: " + classes.length);
214         }
215 
216         Class clazz = getMergedClass(injector, classes, prefixes, observerMode);
217 
218         try {
219             if ((observerMode & OBSERVER_ENABLED) == 0)
220                 return clazz.getConstructor(classes);
221             else {
222                 ArrayList classList = new ArrayList(classes.length + 1);
223                 classList.add(InvocationEventObserver.class);
224                 for (int i = 0; i < classes.length; i++)
225                     classList.add(classes[i]);
226                 return clazz.getConstructor((Class[]) classList.toArray(new Class[classList.size()]));
227             }
228         }
229         catch (NoSuchMethodException e) {
230             throw new InternalError(e.toString());
231         }
232     }

382      * Just create the bytecode for the merged class, but don't load it. Since
383      * no ClassInjector is provided to resolve name conflicts, the class name
384      * must be manually provided.  An optional implementation of the
385      * MethodInvocationEventObserver interface may be supplied.
386      *
387      * @param className name to give to merged class
388      * @param classes Source classes used to derive merged class
389      * @param prefixes Optional prefixes to apply to methods of each generated
390      * class to eliminate duplicate method names
391      * @param observeMethods boolean Enable function call profiling.
392      */
393     public static ClassFile buildClassFile(String className,
394                                            Class[] classes, String[] prefixes,
395                                            int observerMode)
396         throws IllegalArgumentException
397     {
398         ClassEntry[] classEntries = new ClassEntry[classes.length];
399         for (int i=0; i<classes.length; i++) {
400             if (prefixes == null || i >= prefixes.length) {
401                 classEntries[i] = new ClassEntry(classes[i]);
402             }
403             else {
404                 String prefix = prefixes[i];
405                 if (prefix != null && prefix.length() == 0) {
406                     prefix = null;
407                 }
408                 classEntries[i] = new ClassEntry(classes[i], prefix);
409             }
410         }
411         return buildClassFile(null, className, classEntries, observerMode);
412     }

414     private static Class getMergedClass(ClassInjector injector,
415                                         Class[] classes,
416                                         String[] prefixes,
417                                         int observerMode)
418         throws IllegalArgumentException
419     {
420         ClassEntry[] classEntries = new ClassEntry[classes.length];
421         for (int i=0; i<classes.length; i++) {
422             // Load the classes from the ClassInjector, just like they will be
423             // when the generated class is resolved.
424             try {
425                 classes[i] = injector.loadClass(classes[i].getName());
426             }
427             catch (ClassNotFoundException e) {
428                 throw new IllegalArgumentException
429                     ("Unable to load class from injector: " + classes[i]);
430             }
431 
432             if (prefixes == null || i >= prefixes.length) {
433                 classEntries[i] = new ClassEntry(classes[i]);
434             }
435             else {
436                 String prefix = prefixes[i];
437                 if (prefix != null && prefix.length() == 0) {
438                     prefix = null;
439                 }
440                 classEntries[i] = new ClassEntry(classes[i], prefix);
441             }
442         }
443 
444         return getMergedClass(injector, classEntries, observerMode);
445     }

447     /**
448      * Creates a class with a public constructor whose parameter types match
449      * the given source classes. Another constructor is also created that
450      * accepts an InstanceFactory.
451      */
452     private static synchronized Class getMergedClass(ClassInjector injector,
453                                                      ClassEntry[] classEntries,
454                                                      int observerMode)
455         throws IllegalArgumentException
456     {
457         Map classListMap = (Map)cMergedMap.get(injector);
458         if (classListMap == null) {
459             classListMap = new HashMap(7);
460             cMergedMap.put(injector, classListMap);
461         }
462 
463         Object key = generateKey(classEntries);
464         String mergedName = (String)classListMap.get(key);
465         if (mergedName != null) {
466             try {
467                 return injector.loadClass(mergedName);
468             }
469             catch (ClassNotFoundException e) {
470             }
471         }
472 
473         ClassFile cf;
474         try {
475             cf = buildClassFile(injector, null, classEntries, observerMode);
476         }
477         catch (IllegalArgumentException e) {
478             e.fillInStackTrace();
479             throw e;
480         }
481 
482         /*
483         try {
484             java.io.FileOutputStream out =
485                 new java.io.FileOutputStream(cf.getClassName() + ".class");
486             cf.writeTo(out);
487             out.close();
488         }
489         catch (Exception e) {
490             e.printStackTrace();
491         }
492         */
493 
494         try {
495             OutputStream stream = injector.getStream(cf.getClassName());
496             cf.writeTo(stream);
497             stream.close();
498         }
499         catch (IOException e) {
500             throw new InternalError(e.toString());
501         }
502 
503         Class merged;
504         try {
505             merged = injector.loadClass(cf.getClassName());
506         }
507         catch (ClassNotFoundException e) {
508             throw new InternalError(e.toString());
509         }
510 
511         classListMap.put(key, merged.getName());
512         return merged;
513     }

515     private static Object generateKey(ClassEntry[] classEntries) {
516         int length = classEntries.length;
517         Object[] mainElements = new Object[length];
518         for (int i=0; i<length; i++) {
519             ClassEntry classEntry = classEntries[i];
520             mainElements[i] = new MultiKey(new Object[] {
521                 classEntry.getClazz().getName(),
522                 classEntry.getMethodPrefix()
523             });
524         }
525         return new MultiKey(mainElements);
526     }

528     /**
529      * @param className if null, auto selects a unique name against injector
530      */
531     private static ClassFile buildClassFile(ClassInjector injector,
532                                             String className,
533                                             ClassEntry[] classEntries,
534                                             int observerMode)
535         throws IllegalArgumentException
536     {
537         Set classSet = new UsageSet(classEntries.length * 2 + 1);
538         Set nonConflictingClasses = new HashSet(classEntries.length * 2 + 1);
539         String commonPackage = null;
540         Map methodMap = new HashMap();
541 
542 
543         for (int i=0; i<classEntries.length; i++) {
544             ClassEntry classEntry = classEntries[i];
545             Class clazz = classEntry.getClazz();
546 
547             if (clazz.isPrimitive()) {
548                 throw new IllegalArgumentException
549                     ("Merged classes cannot be primitive: " + clazz);
550             }
551 
552             if (!classSet.add(classEntry)) {
553                 throw new IllegalArgumentException
554                     ("Class is specified more than once: " + clazz);
555             }
556 
557             if (!Modifier.isPublic(clazz.getModifiers())) {
558                 String classPackage = clazz.getName();
559                 int index = classPackage.lastIndexOf('.');
560                 if (index < 0) {
561                     classPackage = "";
562                 }
563                 else {
564                     classPackage = classPackage.substring(0, index);
565                 }
566 
567                 if (commonPackage == null) {
568                     commonPackage = classPackage;
569                 }
570                 else if (!commonPackage.equals(classPackage)) {
571                     throw new IllegalArgumentException
572                         ("Not all non-public classes defined in same " +
573                          "package: " + commonPackage);
574                 }
575             }
576 
577             // Innocent until proven guilty.
578             nonConflictingClasses.add(classEntry);
579 
580             Method[] methods = clazz.getMethods();
581             String prefix = classEntry.getMethodPrefix();
582 
583             for (int j=0; j<methods.length; j++) {
584 
585                 Method method = methods[j];
586                 String name = method.getName();
587                 if ("getObserverMode".equals(name) || "getInvocationObserver".equals(name))
588                     continue; // don't wrap this.
589 
590                 // Workaround for JDK1.2 bug #4187388.
591                 if ("<clinit>".equals(name)) {
592                     continue;
593                 }
594 
595                 MethodEntry methodEntry = new MethodEntry(clazz, method, name);
596                 MethodEntry existing = (MethodEntry)methodMap.get(methodEntry);
597 
598                 if (existing == null) {
599                     methodMap.put(methodEntry, methodEntry);
600                 }
601                 else if (existing.returnTypeDiffers(methodEntry)) {
602                     nonConflictingClasses.remove(classEntry);
603                     if (prefix == null) {
604                         throw new IllegalArgumentException
605                             ("Conflicting return types: " +
606                              existing + ", " + methodEntry);
607                     }
608                 }
609 
610                 if (prefix != null) {
611                     name = prefix + name;
612 
613                     methodEntry = new MethodEntry(clazz, method, name);
614                     existing = (MethodEntry)methodMap.get(methodEntry);
615 
616                     if (existing == null) {
617                         methodMap.put(methodEntry, methodEntry);
618                     }
619                     else if (existing.returnTypeDiffers(methodEntry)) {
620                         nonConflictingClasses.remove(classEntry);
621                         throw new IllegalArgumentException
622                             ("Conflicting return types: " +
623                              existing + ", " + methodEntry);
624                     }
625                 }
626             }
627         }
628 
629         if (className == null) {
630             int id = 0;
631             Iterator it = classSet.iterator();
632             while (it.hasNext()) {
633                 id = id * 31 + it.next().hashCode();
634             }
635 
636             className = "MergedClass$";
637             try {
638                 while (true) {
639                     className = "MergedClass$" + (id & 0xffffffffL);
640                     if (commonPackage != null && commonPackage.length() > 0) {
641                         className = commonPackage + '.' + className;
642                     }
643                     try {
644                         injector.loadClass(className);
645                     }
646                     catch (LinkageError e) {
647                     }
648                     id++;
649                 }
650             }
651             catch (ClassNotFoundException e) {
652             }
653         }
654 
655         ClassFile cf = new ClassFile(className);
656         cf.getModifiers().setFinal(true);
657         cf.markSynthetic();
658 
659         for (int i=0; i<classEntries.length; i++) {
660             ClassEntry classEntry = classEntries[i];
661             if (nonConflictingClasses.contains(classEntry)) {
662                 addAllInterfaces(cf, classEntry.getClazz());
663             }
664         }
665 
666         Modifiers privateAccess = new Modifiers();
667         privateAccess.setPrivate(true);
668         Modifiers privateFinalAccess = new Modifiers();
669         privateFinalAccess.setPrivate(true);
670         privateFinalAccess.setFinal(true);
671         Modifiers publicAccess = new Modifiers();
672         publicAccess.setPublic(true);
673         Modifiers privateStaticAccess = new Modifiers();
674         privateStaticAccess.setPrivate(true);
675         privateStaticAccess.setStatic(true);
676         privateStaticAccess.setStatic(true);
677         Modifiers publicStaticAccess = new Modifiers();
678         publicStaticAccess.setPublic(true);
679         publicStaticAccess.setStatic(true);
680

807         // observer can be optionally added.

811         else

819         builder.storeField("instanceFactory", instanceFactoryType);

821             builder.loadThis();
822             builder.loadLocal(builder.getParameters()[1]);
823             builder.storeStaticField("mInvocationObserver", methodObserverType);
824         }

826         builder.returnVoid();

830

832         for (int i=0; i<classEntries.length; i++) {
833             ClassEntry classEntry = classEntries[i];

835             String fieldName = fieldNames[i];
836             TypeDesc type = types[i];
837

843                 if ("<clinit>".equals(method.getName())) {

845                 }

853                 }

857                         (clazz, method, prefix + method.getName());

859                         methodSet.remove(methodEntry);
860                         addWrapperMethod(cf, methodEntry, fieldName, type, (observerMode & OBSERVER_ENABLED) != 0 && (observerMode & OBSERVER_ACTIVE) != 0);
861                     }
862                 }

864         }

886     private static GenericTypeDesc resolve(GenericType type) {
887         Type genericType = type.getGenericType();
888         if (genericType instanceof Class) {
889             return GenericTypeFactory.fromType(genericType);
890         }
891         else if (genericType instanceof TypeVariable) {
892             return resolve(type.getRawType());
893         }
894         else if (genericType instanceof ParameterizedType) {
895             GenericType[] args = type.getTypeArguments();
896             GenericTypeDesc[] argDescs = new GenericTypeDesc[args.length];
897             for (int i = 0; i < args.length; i++) {
898                 argDescs[i] = resolve(args[i]);
899             }
900 
901             return ParameterizedTypeDesc.forType
902             (
903                 (ClassTypeDesc) GenericTypeFactory.fromType
904                 (
905                     type.getRawType().getType()
906                 ),
907                 argDescs
908             );
909         }
910         else if (genericType instanceof WildcardType) {
911             return resolve(type.getRawType());
912         }
913         else if (genericType instanceof GenericArrayType) {
914             GenericTypeDesc compType =
915                 resolve(type.getRootComponentType());
916 
917             for (int i =  0; i < type.getDimensions(); i++) {
918                 compType = GenericArrayTypeDesc.forType(compType);
919             }
920 
921             return compType;
922         }
923         else {
924             throw new IllegalStateException("invalid type: " + type);
925         }

928     private static void addWrapperMethod(ClassFile cf,
929                                          MethodEntry methodEntry,
930                                          String fieldName,
931                                          TypeDesc type,
932                                          boolean observeMethods) {
933         Class methodType = methodEntry.getType();
934         Method method = methodEntry.getMethod();
935         TypeDesc methodObserverType =
936             TypeDesc.forClass(InvocationEventObserver.class);
937 
938         // Don't override any methods in Object, especially final ones.
939         if (isDefinedInObject(method)) {
940             return;
941         }
942         
943         Modifiers modifiers = new Modifiers(method.getModifiers());
944         modifiers.setAbstract(false);
945         modifiers.setFinal(true);
946         modifiers.setSynchronized(false);
947         modifiers.setNative(false);
948         modifiers.setStatic(false);
949 
950         Modifiers staticModifiers = (Modifiers)modifiers.clone();
951         staticModifiers.setStatic(true);
952 
953         GenericType returnType =
954             new GenericType(new GenericType(methodType), method.getReturnType(),
955                             method.getGenericReturnType());
956 
957         TypeDesc ret = TypeDesc.forClass(returnType.getRawType().getType(),
958                                          resolve(returnType));
959 
960         Class[] paramClasses = method.getParameterTypes();
961         Type[] paramTypes = method.getGenericParameterTypes();
962         TypeDesc[] params = new TypeDesc[paramClasses.length];
963         for (int i=0; i<params.length; i++) {
964             GenericType paramType =
965                 new GenericType(new GenericType(methodType), paramClasses[i],
966                                 paramTypes[i]);
967 
968             params[i] = TypeDesc.forClass(paramType.getRawType().getType(),
969                                           resolve(paramType));
970         }
971         
972         MethodInfo mi;
973         if (Modifier.isStatic(method.getModifiers())) {
974             mi = cf.addMethod
975                 (staticModifiers, methodEntry.getName(), ret, params);
976         }
977         else {
978             mi = cf.addMethod(modifiers, methodEntry.getName(), ret, params);
979         }
980 
981         // Exception stuff...
982         Class[] exceptions = method.getExceptionTypes();
983         for (int i=0; i<exceptions.length; i++) {
984             mi.addException(exceptions[i].getName());
985         }
986 
987         // Delegate to wrapped object.
988         CodeBuilder builder = new CodeBuilder(mi);
989 
990         LocalVariable retVal = null;
991         if (method.getReturnType() != void.class) {
992             retVal = builder.createLocalVariable("retVal", ret);
993         }
994 
995         LocalVariable startTime = null;
996         if (observeMethods) {
997             startTime = builder.createLocalVariable("startTime",
998                 TypeDesc.forClass(long.class));
999             builder.loadStaticField("mInvocationObserver", methodObserverType);
1000             builder.invokeInterface(methodObserverType.getFullName(), "currentTime",
1001                 TypeDesc.forClass(long.class));
1002             builder.storeLocal(startTime);
1003 
1004         }
1005 
1006         if (!Modifier.isStatic(method.getModifiers())) {
1007             builder.loadThis();
1008             builder.loadField(fieldName, type);
1009             Label isNonNull = builder.createLabel();
1010             builder.dup();
1011             builder.ifNullBranch(isNonNull, false);
1012             // Discard null field value.
1013             builder.pop();
1014             // Call the field access method, which in turn calls the
1015             // InstanceFactory.
1016             builder.loadThis();
1017             builder.invokePrivate(fieldName, type);
1018             isNonNull.setLocation();
1019         }
1020         LocalVariable[] locals = builder.getParameters();
1021         for (int i=0; i<locals.length; i++) {
1022             builder.loadLocal(locals[i]);
1023         }
1024 
1025         builder.invoke(method);
1026 
1027         if (method.getReturnType() != void.class)
1028             builder.storeLocal(retVal);
1029 
1030 
1031         if (observeMethods) {
1032             builder.loadStaticField("mInvocationObserver", methodObserverType);
1033             builder.loadConstant(null);
1034             builder.loadConstant(methodEntry.getName());
1035             builder.loadStaticField("mInvocationObserver", methodObserverType);
1036             builder.invokeInterface(methodObserverType.getFullName(), "currentTime",
1037                 TypeDesc.forClass(long.class));
1038             builder.loadLocal(startTime);
1039             builder.math(Opcode.LSUB);
1040             builder.invokeInterface(methodObserverType.getFullName(), "invokedEvent", null,
1041                 new TypeDesc[] { TypeDesc.forClass(String.class), TypeDesc.forClass(String.class),
1042                 TypeDesc.forClass(long.class) });
1043         }
1044 
1045         if (method.getReturnType() == void.class) {
1046             builder.returnVoid();
1047         }
1048         else {
1049             builder.loadLocal(retVal);
1050             builder.returnValue(ret);
1051         }
1052     }

1054     private static boolean isDefinedInObject(Method method) {
1055         if (method.getDeclaringClass() == Object.class) {
1056             return true;
1057         }
1058 
1059         Class[] types = method.getParameterTypes();
1060         String name = method.getName();
1061 
1062         if (types.length == 0) {
1063             return
1064                 "hashCode".equals(name) ||
1065                 "clone".equals(name) ||
1066                 "toString".equals(name) ||
1067                 "finalize".equals(name);
1068         }
1069         else {
1070             return
1071                 types.length == 1 &&
1072                 types[0] == Object.class &&
1073                 "equals".equals(name);
1074         }
