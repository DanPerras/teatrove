232     /**
233      * @exception IllegalArgumentException if opcode is invalid
234      */
235     public final static String getMnemonic(byte opcode) 
236         throws IllegalArgumentException {
237         try {
238             return Mnemonic.m[opcode & 0xff];
239         }
240         catch (ArrayIndexOutOfBoundsException e) {
241             throw new IllegalArgumentException
242                 ("Opcode not valid: " + opcode);
243         }
244     }

246     /**
247      * Reverses the condition for an "if" opcode. i.e. IFEQ is changed to
248      * IFNE.
249      */
250     public final static byte reverseIfOpcode(byte opcode) {
251         // Actually, because the numbers assigned to the "if" opcodes
252         // were so cleverly chosen, all I really need to do is toggle
253         // bit 0. I'm not going to do that because I still need to check if 
254         // an invalid opcode was passed in.
255 
256         switch (opcode) {
257         case IF_ACMPEQ:
258             return IF_ACMPNE;
259         case IF_ACMPNE:
260             return IF_ACMPEQ;
261         case IF_ICMPEQ:
262             return IF_ICMPNE;
263         case IF_ICMPNE:
264             return IF_ICMPEQ;
265         case IF_ICMPLT:
266             return IF_ICMPGE;
267         case IF_ICMPGE:
268             return IF_ICMPLT;
269         case IF_ICMPGT:
270             return IF_ICMPLE;
271         case IF_ICMPLE:
272             return IF_ICMPGT;
273         case IFEQ:
274             return IFNE;
275         case IFNE:
276             return IFEQ;
277         case IFLT:
278             return IFGE;
279         case IFGE:
280             return IFLT;
281         case IFGT:
282             return IFLE;
283         case IFLE:
284             return IFGT;
285         case IFNONNULL:
286             return IFNULL;
287         case IFNULL:
288             return IFNONNULL;
289         default:
290             throw new IllegalArgumentException
291                 ("Opcode not an if instruction: " + getMnemonic(opcode));
292         }
293     }
