67          addMethods(methods, iface, types, parent);

97      /**
98       * Get the number of dimensions for the specified type.  If the type is
99       * a {@link GenericArrayType}, the dimensions will equal the number of
100      * actual dimensions.  Otherwise, 0 will be returned.
101      * 
102      * @param type  The underlying type
103      * 
104      * @return  The number of dimensions
105      */
106     private static int getDimensions(Type type) {
107         int dimensions = 0;
108         while (type instanceof GenericArrayType) {
109             dimensions++;
110             type = ((GenericArrayType) type).getGenericComponentType();
111         }
112         
113         return dimensions;
114     }

116     /**
117      * Get the {@link TypeVariable} instance for the specified {@link Type}.
118      * If the type is not a {@link TypeVariable}, then <code>null</code> is
119      * returned.  If the type is a {@link GenericArrayType}, its root type is
120      * looked up and then checked.
121      * 
122      * @param type  The associated type
123      * 
124      * @return  The type variable instance or <code>null</code>
125      */
126     private static TypeVariable<?> getTypeVariable(Type type) {
127         // find root type if generic array
128         while (type instanceof GenericArrayType) {
129             type = ((GenericArrayType) type).getGenericComponentType();
130         }
131         
132         // return type variable if valid
133         if (type instanceof TypeVariable) {
134             return (TypeVariable<?>) type;
135         }
136         
137         // non-type variable, return null
138         return null;
139     }

141     /**
142      * Check whether a bridge is required for the given method and generic
143      * type information.  If the return type or any parameter contains a type
144      * variable that is specified and overridden with a more specific type
145      * via the given types, then a bridge will be required.
146      * 
147      * @param types  The set of named type variables
148      * @param method  The method to check
149      * 
150      * @return  The state of whether a bridge is required
151      */
152     private static boolean isBridgeRequired(Map<String, Class<?>> types,
153                                             Method method) {
154         // check return type
155         TypeVariable<?> var = getTypeVariable(method.getGenericReturnType());
156         if (var != null && types.containsKey(var.getName())) {
157             return true;
158         }
159         
160         // check param types
161         for (Type paramType : method.getGenericParameterTypes()) {
162             var = getTypeVariable(paramType);
163             if (var != null && types.containsKey(var.getName())) {
164                 return true;
165             }            
166         }
167         
168         // not required
169         return false;
170     }

173      * Find the specified method on the given class.  If not specified, then
174      * <code>null</code> is returned.  Otherwise, the associated method is
175      * returned.  The specified method definition does not need to be from the
176      * specified class.  This method is usually used to check a method defined
177      * in an interface or superclass on another class within the hierarchy.
178      * 
179      * @param clazz  The class to check
180      * @param method  The method definition
181      * 
182      * @return The associated method if existant, otherwise <code>null</code>
183      */
184     private static Method findMethod(Class<?> clazz, Method method) {
185         try {
186             return clazz.getMethod(method.getName(), 
187                                    method.getParameterTypes());
188         }
189         catch (NoSuchMethodException e) {
190             return null;
191         }
192     }

194     /**
195      * Get the root class type based on the given type and set of named type
196      * variables.  If the type is already a Class, it is renamed as is.  If the
197      * type is a parameterized type, then the raw type is returned.  If the
198      * type is a wildcard, the lower bounds is returned.  Otherwise, if the 
199      * type is a type variable, its associated value from the types is returned.
200      * 
201      * @param types  The named type variales to pass thru
202      * @param type  The associated type
203      * 
204      * @return  The underlying root class for the type
205      */
206     private static final Class<?> getRootType(Map<String, Class<?>> types,
207                                               Type type) {
208         
209         // if type explicitly defined, pass thru as is
210         if (type instanceof Class) {
211             return (Class<?>) type;
212         }
213         
214         // if type defined as parameter, pass thru root type
215         else if (type instanceof ParameterizedType) {
216             Type raw = ((ParameterizedType) type).getRawType();
217             return getRootType(types, raw);
218         }
219         
220         // if type defined as wildcard, pass thru bounds
221         // NOTE: this does not support ? super X format
222         else if (type instanceof WildcardType) {
223             Type[] bounds = ((WildcardType) type).getUpperBounds();
224             if (bounds.length >= 1) {
225                 return getRootType(types, bounds[0]);
226             }
227         }
228         
229         // otherwise, either generic array or plain type variable
230         // lookup in types and return specified type if provided
231         else {
232             TypeVariable<?> raw = getTypeVariable(type);
233             if (raw != null) {
234                 // verify type is overridden
235                 String name = raw.getName();
236                 if (types.containsKey(name)) {
237                     // get actual type and convert to array if necessary
238                     Class<?> actual = types.get(name);
239                     int dimensions = getDimensions(type);
240                     if (dimensions >= 1) {
241                         actual = Array.newInstance
242                         (
243                             actual, dimensions
244                         ).getClass();
245                     }
246                     
247                     return actual;
248                 }
249             }
250         }
251         
252         // nothing found
253         return null;
254     }

256     /**
257      * Get the actual type for the given type.  If the type is a type variable
258      * that is specified in the given types, then the associated class is
259      * returned.  Otherwise, the specified class is returned.
260      * 
261      * @param types  The set of named type variables
262      * @param type  The generic type of the method
263      * @param clazz  The underlying type of the method
264      * 
265      * @return  The actual type to use in the method
266      */
267     private static Class<?> getMethodType(Map<String, Class<?>> types, 
268                                           Type type, Class<?> clazz) {
269         
270         // check if type is an actual type variable
271         TypeVariable<?> variable = getTypeVariable(type);
272         if (variable != null) {
273             
274             // verify type is overridden
275             String name = variable.getName();
276             if (types.containsKey(name)) {
277                 
278                 // convert to array if necessary
279                 Class<?> actual = types.get(name);
280                 int dimensions = getDimensions(type);
281                 if (dimensions > 0) {
282                     actual = Array.newInstance(actual, dimensions).getClass();
283                 }
284                 
285                 // return overridden type
286                 return actual;
287             }
288         }
289         
290         // nothing overriden, so just return base class
291         return clazz;
292     }

294     /**
295      * Add a method to the set of existing methods.  If the method is not
296      * bridged, it will overwrite any existing signature.  Otherwise, it will
297      * only update the set if not previously defined.  Note that the signature
298      * of the method entry includes the method name, parameter types, and the
299      * return type.
300      * 
301      * @param methods  The set of methods
302      * @param method  The method to add
303      */
304     private static void addMethod(Set<MethodEntry> methods, 
305                                   MethodEntry method) {
306         if (!method.isBridged()) {
307             methods.remove(method);
308             methods.add(method);
309         }
310         else if (!methods.contains(method)) {
311             methods.add(method);
312         }
313     }

315     /**
316      * Recursive method that adds all applicable methods for the given class,
317      * which is expected to be an interface.  The tree is walked for the class
318      * and all super interfaces in recursive fashion to get each available 
319      * method.  Any method that is already defined by the given parent will be
320      * ignored.  Otherwise, the method, including any required bridged methods
321      * will be added.  The specified list of types define the optional type
322      * parameters of the given interface class.  These types will propogate up
323      * the tree as necessary for any super-interfaces that match.
324      * 
325      * @param methods  The set of existing methods
326      * @param clazz  The current interface class
327      * @param types  The set of named type variable
328      * @param parent  The optional parent class
329      */
330     private static void addMethods(Set<MethodEntry> methods, 
331                                    Class<?> clazz, Map<String, Class<?>> types,
332                                    Class<?> parent) {
333         
334         // validate interface
335         if (!clazz.isInterface()) { 
336             throw new IllegalStateException("class must be interface: " + clazz);
337         }
338         
339         // loop through each method (only those declared)
340         for (Method method : clazz.getDeclaredMethods()) {
341             addMethods(methods, clazz, types, parent, method);
342         }
343 
344         // recursively add each super interface providing type info if valid
345         addParentMethods(methods, clazz, types, parent, null);
346     }

348     /**
349      * Recursive method that adds all applicable methods related to the given 
350      * class, which is expected to be an interface.  The tree is walked for the 
351      * class and all super interfaces in recursive fashion to get each available 
352      * method that matches the provided method.  Any method that is already 
353      * defined by the given parent will be ignored.  Otherwise, the method, 
354      * including any required bridged methods will be added.  The specified list 
355      * of types define the optional type parameters of the given interface 
356      * class.  These types will propogate up the tree as necessary for any 
357      * super-interfaces that match.
358      * 
359      * @param methods  The set of existing methods
360      * @param clazz  The current interface class
361      * @param types  The set of named type variable
362      * @param parent  The optional parent class
363      * @param method  The method to implement
364      */
365     private static void addMethods(Set<MethodEntry> methods, 
366                                    Class<?> clazz, Map<String, Class<?>> types,
367                                    Class<?> parent, Method method) {
368         
369         // validate interface
370         //if (!clazz.isInterface()) { 
371         //    throw new IllegalStateException("class must be interface: " + clazz);
372         //}
373         
374         // check if the method requires a bridge and add generic method
375         // with overridden type params
376         MethodEntry impl = null;
377         boolean bridged = isBridgeRequired(types, method);
378         
379         // do not add method if already implemented
380         // ignore if method already defined on the parent
381         if (!bridged && parent != null && findMethod(parent, method) != null) {
382             return;
383         }
384         
385         // bridge if necessary
386         if (bridged) {
387             // get the underying generic type if provided
388             Class<?> returnType = getMethodType
389             (
390                 types, method.getGenericReturnType(), method.getReturnType()
391             );
392             
393             // get the underlying generic params if provided
394             Type[] paramTypes = method.getGenericParameterTypes();
395             Class<?>[] paramClasses = method.getParameterTypes();
396             for (int i = 0; i < paramClasses.length; i++) {
397                 paramClasses[i] = getMethodType
398                 (
399                     types, paramTypes[i], paramClasses[i]
400                 );
401             }
402             
403             // check if only return type is overridden and only add impl
404             // if not yet defined...this is possible since return type is
405             // not part of a method signature for equality purposes so we
406             // only need to implement in a concrete implementation once and
407             // bridge all other methods.  Parameters, on the other hand
408             // must always be explicitly implemented to satisfy their
409             // interface definition.
410             MethodEntry bridgeMethod = null;
411             if (Arrays.equals(paramClasses, method.getParameterTypes())) {
412                 for (MethodEntry mentry : methods) {
413                     if (mentry.getName().equals(method.getName()) &&
414                         Arrays.equals(mentry.getParamTypes(), 
415                                       method.getParameterTypes())) {
416                         bridgeMethod = mentry;
417                     }
418                 }
419             }
420             
421             // add implementation
422             impl = new MethodEntry(
423                 method.getName(), returnType, paramClasses, bridgeMethod
424             );
425             addMethod(methods, impl);
426         }
427         
428         // add actual method definition, bridging if implemented above
429         addMethod(methods, new MethodEntry(method.getName(),
430                                            method.getReturnType(), 
431                                            method.getParameterTypes(),
432                                            impl));
433         
434         // recursively add each super interface providing type info if valid
435         addParentMethods(methods, clazz, types, parent, method);
436     }

438     /**
439      * Recursively add all methods walking the interface hiearchy for the given
440      * class.
441      * 
442      * @param methods  The set of existing methods
443      * @param clazz  The current interface class
444      * @param types  The set of named type variable
445      * @param parent  The optional parent class
446      * @param method  The method to implement
447      */
448     private static void addParentMethods(Set<MethodEntry> methods, 
449                                          Class<?> clazz, 
450                                          Map<String, Class<?>> types,
451                                          Class<?> parent, Method method) {
452         // recursively add each super interface providing type info if valid
453         Class<?>[] ifaces = clazz.getInterfaces();
454         Type[] generics = clazz.getGenericInterfaces();
455         for (int i = 0; i < ifaces.length; i++) {
456             Class<?> iface = ifaces[i];
457             Type generic = iface;
458             if (generics != null && i < generics.length) {
459                 generic = generics[i];
460             }
461             
462             // check if interface is parameterized and lookup type information
463             // cascading current type information if types match
464             if (generic instanceof ParameterizedType) {
465                 TypeVariable<?>[] vars = iface.getTypeParameters();
466                 ParameterizedType ptype = (ParameterizedType) generic;
467                 Type[] arguments = ptype.getActualTypeArguments();
468                 
469                 // check each parameter for specific type 
470                 Map<String, Class<?>> args = new HashMap<String, Class<?>>();
471                 for (int j = 0; j < vars.length; j++) {
472                     Type arg = arguments[j];
473                     TypeVariable<?> var = vars[j];
474 
475                     // check if root type defined and pass type thru
476                     Class<?> root = getRootType(types, arg);
477                     if (root != null) {
478                         args.put(var.getName(), root);
479                     }
480                 }
481                 
482                 // recursively invoke
483                 if (method == null) {
484                     addMethods(methods, (Class<?>) ptype.getRawType(), args, 
485                                parent);
486                 }
487                 else {
488                     addMethods(methods, (Class<?>) ptype.getRawType(), args, 
489                                parent, method);
490                 }
491             }
492             
493             // normal class, so recursively invoke as is
494             else if (iface instanceof Class<?>) {
495                 if (method == null) {
496                     addMethods(methods, iface, new HashMap<String, Class<?>>(),
497                                parent);
498                 }
499                 else {
500                     addMethods(methods, iface, new HashMap<String, Class<?>>(),
501                                parent, method);
502                 }
503             }
504         }
505     }

507     /**
508      * Method description containing the method definition and the bridge
509      * definition, if valid.
510      */
511     public static class MethodEntry {
512         private String name;
513         private Class<?> returnType;
514         private Class<?>[] paramTypes;
515         private MethodEntry bridgedMethod;
516         
517         public MethodEntry(String name,
518                            Class<?> returnType, Class<?>[] paramTypes,
519                            MethodEntry bridgedMethod) {
520             this.name = name;
521             this.returnType = returnType;
522             this.paramTypes = paramTypes;
523             this.bridgedMethod = bridgedMethod;
524         }
525         
526         public String getName() { return this.name; }
527         public Class<?> getReturnType() { return this.returnType; }
528         public Class<?>[] getParamTypes() { return this.paramTypes; }
529         public boolean isBridged() { return this.bridgedMethod != null; }
530         public MethodEntry getBridgedMethod() { return this.bridgedMethod; }
531         
532         public boolean equals(Object object) {
533             if (object == this) { return true; }
534             else if (!(object instanceof MethodEntry)) { return false; }
535             
536             MethodEntry entry = (MethodEntry) object;
537             if (!this.getName().equals(entry.getName()) ||
538                 !this.getReturnType().equals(entry.getReturnType()) ||
539                 !Arrays.equals(this.getParamTypes(), entry.getParamTypes())) {
540                 
541                 return false; 
542             }
543             
544             return true;
545         }
546         
547         public int hashCode() {
548             return 13 + 
549                 (19 * this.name.hashCode()) +
550                 (21 * this.returnType.hashCode()) + 
551                 (23 * Arrays.hashCode(this.paramTypes));
552         }
553         
554         public String toString() {
555             StringBuilder buffer = new StringBuilder();
556             buffer.append(this.returnType.getName());
557             buffer.append(" " + this.name + "(");
558             for (int i = 0; i < this.paramTypes.length; i++) {
559                 if (i > 0) { buffer.append(", "); }
560                 buffer.append(this.paramTypes[i].getName());
561             }
562             
563             buffer.append(")");
564             
565             if (this.bridgedMethod != null) {
566                 buffer.append(" [bridge] ")
567                       .append(this.bridgedMethod.toString());
568             }
569             
570             return buffer.toString();
571         }
572     }
