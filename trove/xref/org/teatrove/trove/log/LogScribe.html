17  package org.teatrove.trove.log;
18  
19  import java.io.*;
20  import java.text.*;
21  import java.util.TimeZone;
22  import org.teatrove.trove.util.FastDateFormat;
23  
24  /**
25   * LogScribe is a LogListener that writes log messages to a PrintWriter. Each
26   * message is printed to a line that is prepended with other LogEvent
27   * information. The default prepend format is as follows:
28   *
29   * <pre>
30   * [event type code],[date & time],[thread name],[log source name]>[one space]
31   * </pre>
32   *
33   * The event type codes are <tt>" D", " I", "*W" and "*E"</tt> for debug,
34   * info, warn and error, respectively. The default date format looks like
35   * this: <tt>"1999/06/08 18:08:34.067 PDT"</tt>. If there is no log source, or
36   * it has no name, it is omitted from the prepend. Here is a sample line that
37   * is written out:
38   *
39   * <pre>
40   *  I,1999/06/08 18:08:34.67 PDT,main> Started Transaction Manager
41   * </pre>
42   *
43   * @author Brian S O'Neill
44   */
45  public class LogScribe implements LogListener {
46      private PrintWriter mWriter;
47      private DateFormat mSlowFormat;
48      private FastDateFormat mFastFormat;
49  
50      private boolean mShowThread = true;
51      private boolean mShowSourceName = true;
52  
53      public LogScribe(PrintWriter writer) {
54          this(writer, (FastDateFormat)null);
55      }
56  
57      public LogScribe(PrintWriter writer, DateFormat format) {
58          mWriter = writer;
59          mSlowFormat = format;
60          if (format == null) {
61              mFastFormat =
62                  FastDateFormat.getInstance("yyyy/MM/dd HH:mm:ss.SSS z");
63          }
64          else if (format instanceof SimpleDateFormat) {
65              SimpleDateFormat simple = (SimpleDateFormat)format;
66              String pattern = simple.toPattern();
67              TimeZone timeZone = simple.getTimeZone();
68              DateFormatSymbols symbols = simple.getDateFormatSymbols();
69              mFastFormat =
70                  FastDateFormat.getInstance(pattern, timeZone, null, symbols);
71          }
72      }
73  
74      public LogScribe(PrintWriter writer, FastDateFormat format) {
75          mWriter = writer;
76          mSlowFormat = null;
77          if (format == null) {
78              format = FastDateFormat.getInstance("yyyy/MM/dd HH:mm:ss.SSS z");
79          }
80          mFastFormat = format;
81      }
82  
83      public void logMessage(LogEvent e) {
84          String message = e.getMessage();
85          if (message != null) {
86              synchronized (mWriter) {
87                  mWriter.print(createPrepend(e));
88                  mWriter.println(message);
89                  mWriter.flush();
90              }
91          }
92      }
93  
94      public void logException(LogEvent e) {
95          Throwable t = e.getException();
96          
97          if (t == null) {
98              logMessage(e);
99          }
100         else {
101             synchronized (mWriter) {
102                 mWriter.print(createPrepend(e));
103                 t.printStackTrace(mWriter);
104                 mWriter.flush();
105             }
106         }
107     }
108 
109     /**
110      * The showing of the event thread name is on by default.
111      */
112     public boolean isShowThreadEnabled() {
113         return mShowThread;
114     }
115 
116     public void setShowThreadEnabled(boolean enabled) {
117         mShowThread = enabled;
118     }
119 
120     /**
121      * The showing of the event source name is on by default.
122      */
123     public boolean isShowSourceEnabled() {
124         return mShowSourceName;
125     }
126 
127     public void setShowSourceEnabled(boolean enabled) {
128         mShowSourceName = enabled;
129     }
130 
131     /**
132      * Creates the default line prepend for a message.
133      */
134     protected String createPrepend(LogEvent e) {
135         StringBuffer pre = new StringBuffer(80);
136                 
137         String code = "??";
138         switch (e.getType()) {
139         case LogEvent.DEBUG_TYPE:
140             code = " D";
141             break;
142         case LogEvent.INFO_TYPE:
143             code = " I";
144             break;
145         case LogEvent.WARN_TYPE:
146             code = "*W";
147             break;
148         case LogEvent.ERROR_TYPE:
149             code = "*E";
150             break;
151         }
152 
153         pre.append(code);
154         pre.append(',');
155         if (mFastFormat != null) {
156             pre.append(mFastFormat.format(e.getTimestamp()));
157         }
158         else {
159             synchronized (mSlowFormat) {
160                 pre.append(mSlowFormat.format(e.getTimestamp()));
161             }
162         }
163 
164         if (isShowThreadEnabled()) {
165             pre.append(',');
166             pre.append(e.getThreadName());
167         }
168 
169         if (isShowSourceEnabled()) {
170             Log source = e.getLogSource();
171             if (source != null) {
172                 String sourceName = source.getName();
173                 if (sourceName != null) {
174                     pre.append(',');
175                     pre.append(sourceName);
176                 }
177             }
178         }
179 
180         pre.append('>');
181         pre.append(' ');
182 
183         return pre.toString();
184     }
