30  /**
31   * A CharToByteBuffer that keeps track of interned strings (mainly string
32   * literals) and statically caches the results of those strings after applying
33   * a byte conversion. This can improve performance if many of the strings being
34   * passed to the append method have been converted before.
35   *
36   * @author Brian S O'Neill
37   */
38  public class InternedCharToByteBuffer

45      private static Map cEncodings = new HashMap(7);
46  
47      private static Random cLastRandom = new Random();

49      private static Map getConvertedCache(String encoding) {
50          synchronized (cEncodings) {
51              Map cache = (Map)cEncodings.get(encoding);
52              if (cache == null) {
53                  cache = Collections.synchronizedMap(new IdentityMap());
54                  cEncodings.put(encoding, cache);
55              }
56              return cache;
57          }
58      }
59  
60      private static Random getRandom() {
61          synchronized (cLastRandom) {
62              return cLastRandom = new Random(cLastRandom.nextLong());
63          }
64      }
