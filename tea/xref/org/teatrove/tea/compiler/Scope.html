30  /**
31   * A Scope encapsulates a set of declared variables and references to them.
32   * Scopes can be nested, and child scopes have access to variables in the 
33   * parent scope.
34   *
35   * @author Brian S O'Neill
36   */
37  public class Scope {
38      // Shared variables, maps Variable objects to Variable objects.
39      private Map<Variable, Variable> mVariables;
40      private Scope mParent;
41      private Collection<Scope> mChildren;
42      
43      // Set of private Variables declared in this scope.
44      private Set<Variable> mPrivateVars;
45  
46      // Maps String variable names to Variable objects for locally available
47      // variables.
48      private Map<String, Variable> mDeclared = new HashMap<String, Variable>(11);

58      public Scope(Scope parent) {
59          if ((mParent = parent) != null) {
60              mVariables = parent.mVariables;
61              if (parent.mChildren == null) {
62                  parent.mChildren = new ArrayList<Scope>(5);
63              }
64              parent.mChildren.add(this);
65          }
66          else {
67              mVariables = new HashMap<Variable, Variable>(53);
68          }
69      }

99      /**
100      * Declare a variable for use in this scope. If no variable of this name
101      * and type has been defined, it is added to the shared set of pooled
102      * variables. Returns the actual Variable object that should be used
103      * instead.
104      *
105      * @param isPrivate when true, variable declaration doesn't leave this
106      * scope during an intersection or promotion
107      */
108     public Variable declareVariable(Variable var, boolean isPrivate) {
109         if (mVariables.containsKey(var)) {
110             var = mVariables.get(var);
111         }
112         else {
113             mVariables.put(var, var);
114         }
115 
116         mDeclared.put(var.getName(), var);
117 
118         if (isPrivate) {
119             if (mPrivateVars == null) {
120                 mPrivateVars = new HashSet<Variable>(7);
121             }
122             mPrivateVars.add(var);
123         }
124         else {
125             if (mPrivateVars != null) {
126                 mPrivateVars.remove(var);
127             }
128         }
129 
130         return var;
131     }

134      * Declare new variables in this scope. Entries in the array are replaced
135      * with actual Variable objects that should be used instead.
136      */
137     public void declareVariables(Variable[] vars) {
138         for (int i=0; i<vars.length; i++) {
139             vars[i] = declareVariable(vars[i]);

154     /**
155      * Returns a declared variable by name. Search begins in this scope and
156      * moves up into parent scopes. If not found, null is returned. A public-
157      * only variable can be requested.
158      *
159      * @return Null if no declared variable found with the given name
160      */
161     public Variable getDeclaredVariable(String name, boolean publicOnly) {
162         //private Set mPrivateVars;
163 
164         Variable var = mDeclared.get(name);
165         if (var != null) {
166             // If its okay to be private or its public then...
167             if (!publicOnly || mPrivateVars == null ||
168                 !mPrivateVars.contains(var)) {
169                 return var;
170             }
171         }
172         
173         if (mParent != null) {
174             return mParent.getDeclaredVariable(name);
175         }
176         
177         return null;
178     }

191      * Attempt to bind variable reference to a variable in this scope or a
192      * parent scope. If the variable to bind to isn't available or doesn't
193      * exist, false is returned.
194      *
195      * @return true if reference has been bound
196      */    
197     public boolean bindToVariable(VariableRef ref) {
198         String name = ref.getName();
199         Variable var = getDeclaredVariable(name);
200         
201         if (var != null) {
202             ref.setType(null);
203             ref.setVariable(var);
204             mVariableRefs.add(ref);
205             return true;
206         }
207         else {
208             return false;
209         }

224     private static void fillVariableRefs(Collection<VariableRef> refs,
225                                          Scope scope) {
226         refs.addAll(scope.mVariableRefs);
227 
228         Collection<Scope> children = scope.mChildren;
229         if (children != null) {
230             Iterator<Scope> it = children.iterator();
231             while (it.hasNext()) {
232                 fillVariableRefs(refs, it.next());
233             }
234         }
235     }

248     /**
249      * Returns all the references made from this scope and all child scopes to
250      * variables declared outside of this scope.
251      */
252     public VariableRef[] getOutOfScopeVariableRefs() {
253         Scope parent;
254         if ((parent = getParent()) == null) {
255             return new VariableRef[0];
256         }
257 
258         Collection<VariableRef> allRefs = new ArrayList<VariableRef>();
259         fillVariableRefs(allRefs, this);
260 
261         Collection<VariableRef> refs =
262             new ArrayList<VariableRef>(allRefs.size());
263 
264         Iterator<VariableRef> it = allRefs.iterator();
265         while (it.hasNext()) {
266             VariableRef ref = it.next();
267             Variable var = ref.getVariable();
268             if (var != null &&
269                 parent.getDeclaredVariable(var.getName()) == var) {
270                 refs.add(ref);
271             }
272         }
273 
274         VariableRef[] refsArray = new VariableRef[refs.size()];
275         return refs.toArray(refsArray);
276     }

330     /**
331      * Returns the intersection of this scope against the one given. An
332      * intersection operates on each scope's locally declared variables,
333      * producing a list of variables that both scopes may inherit.
334      * <p>
335      * The list may contain undeclared variables, and so all returned variables
336      * must be re-declared in a common parent scope. This responsibility is
337      * left to the caller, intersect does not alter the internal state of
338      * either scope.
339      * <p>
340      * This method is designed specifically for combining the locally
341      * declared variables of the "then" and "else" parts of an if statement.
342      *
343      * @return variables representing the intersection
344      */
345     public Variable[] intersect(Scope scope) {
346         Collection<Variable> intersection = new ArrayList<Variable>();
347         
348         // A set of variable names that have been moved into the intersection.
349         Set<String> matchedNames = new HashSet<String>(7);
350 
351         intersectFrom(this, scope, matchedNames, intersection);
352         intersectFrom(scope, this, matchedNames, intersection);
353 
354         Variable[] vars = new Variable[intersection.size()];
355         return intersection.toArray(vars);
356     }

358     /**
359      * Returns variables to promote from this scope to a parent scope.
360      * Promote is similar to intersect, except it operates on this scope and
361      * its parent (if it has one).
362      * <p>
363      * The list may contain undeclared variables, and so all returned variables
364      * must be re-declared in a common parent scope. This responsibility is
365      * left to the caller, promote does not alter the internal state of this
366      * scope or its parent.
367      * <p>
368      * This method is designed specifically for promoting the locally
369      * declared variables of a loop statement's body.
370      *
371      * @return variables to promote
372      */
373     public Variable[] promote() { 
374         Scope parent = getParent();
375         if (parent == null) {
376             return new Variable[0];
377         }
378 
379         Collection<Variable> promotion = new ArrayList<Variable>();
380         
381         // A set of variable names that have been moved into the promotion.
382         Set<String> matchedNames = new HashSet<String>(7);
383 
384         intersectFrom(this, parent, matchedNames, promotion);
385 
386         Variable[] vars = new Variable[promotion.size()];
387         return promotion.toArray(vars);
388     }

390     private static void intersectFrom(Scope scope1, Scope scope2,
391                                       Set<String> matchedNames,
392                                       Collection<Variable> vars) {
393         Set<Variable> privates1 = scope1.mPrivateVars;
394 
395         Variable[] vars1 = scope1.getLocallyDeclaredVariables();
396         for (int i=0; i<vars1.length; i++) {
397             Variable var1 = vars1[i];
398             if (privates1 != null && privates1.contains(var1)) {
399                 continue;
400             }
401 
402             String varName = var1.getName();
403 
404             if (matchedNames.contains(varName)) {
405                 // This variable has already been moved into the intersection.
406                 continue;
407             }
408             else {
409                 matchedNames.add(varName);
410             }
411 
412             Variable var2 = scope2.getDeclaredVariable(varName, true);
413 
414             if (var2 == null) {
415                 // No matching public variable in scope2, so continue.
416                 continue;
417             }
418 
419             Type type1 = var1.getType();
420             Type type2 = var2.getType();
421 
422             // Find common type.
423             Type type = type1.getCompatibleType(type2);
424             if (type == null) {
425                 continue;
426             }
427 
428             // Find a variable to hold common type.
429             Variable var = null;
430             if (type.equals(type1)) {
431                 var = var1;
432             }
433             else if (type.equals(type2)) {
434                 var = var2;
435             }
436             else {
437                 // Create a new variable with the common type.
438                 var = new Variable(var1.getSourceInfo(), varName, type, var != null ? var.isStaticallyTyped() : false);
439             }
440 
441             vars.add(var);
442         }
443     }

455     public String toString() {
456         StringBuffer buf = new StringBuffer();
457         buf.append(super.toString());
458         buf.append('\n');
459         append(buf, this, "");
460         return buf.toString();

463     private void append(StringBuffer buf, Scope scope, String indent) {
464         buf.append(indent);
465         buf.append("{\n");
466 
467         String indentMore = indent + "    ";
468 
469         Variable[] vars = scope.getLocallyDeclaredVariables();
470         for (int i=0; i<vars.length; i++) {
471             Variable var = vars[i];
472 
473             buf.append(indentMore);
474             
475             Set<Variable> privateVars = scope.mPrivateVars;
476             if (privateVars != null && privateVars.contains(var)) {
477                 buf.append("private ");
478             }
479 
480             Type type = var.getType();
481             if (type != null) {
482                 buf.append(type.getFullName());
483             }
484             else {
485                 buf.append("<null>");
486             }
487 
488             buf.append(' ');
489             buf.append(var.getName());
490 
491             buf.append(";  // ");
492             buf.append(var);
493             buf.append('\n');
494         }
495 
496         VariableRef[] refs = scope.getLocalVariableRefs();
497         for (int i=0; i<refs.length; i++) {
498             VariableRef ref = refs[i];
499 
500             buf.append(indentMore);
501             buf.append(ref.getName());
502 
503             buf.append(";  // ");
504             buf.append(ref);
505             buf.append(" to ");
506             buf.append(ref.getVariable());
507             buf.append('\n');
508         }
509         
510         Scope[] children = scope.getChildren();
511         for (int i=0; i<children.length; i++) {
512             append(buf, children[i], indentMore);
513         }
514         
515         buf.append(indent);
516         buf.append("}\n");
517     }
