196           int sourceStartPos,
197           int sourceEndPos,
198           int sourceDetailPos,
199           int tokenID) {
200 
201         mTokenID = tokenID;
202 
203         if (tokenID > LAST_ID) {
204             throw new IllegalArgumentException("Token ID out of range: " +
205                                                tokenID);
206         }
207 
208         if (sourceStartPos == sourceDetailPos) {
209             mInfo = new SourceInfo(sourceLine,
210                                    sourceStartPos, sourceEndPos);
211         }
212         else {
213             mInfo = new SourceDetailedInfo(sourceLine,
214                                            sourceStartPos, sourceEndPos,
215                                            sourceDetailPos);
216         }
217 
218         if (sourceStartPos > sourceEndPos) {
219             // This is an internal error.
220             throw new IllegalArgumentException
221                 ("Token start position greater than end position at line: " +
222                  sourceLine);
223         }
224     }

277     /**
278      * If the given id is a bracket: (,[,{,},],) then the matching bracket's
279      * id is returned. If id is not a bracket, then -1 is returned.
280      */
281     public final static int getMatchingBracket(int id) {
282         if(isOpenBracket(id)) {
283             return ++id;
284         }
285         else if(isCloseBracket(id)) {
286             return --id;
287         }
288 
289         return -1;
290     }

292     /**
293      * If the given StringBuilder starts with a valid token type, its ID is
294      * returned. Otherwise, the token ID UNKNOWN is returned.
295      */
296     public static int findReservedWordID(StringBuilder word) {
297         char c = word.charAt(0);
298 
299         switch (c) {
300         case 'a':
301             if (matches(word, "and")) return AND;
302             if (matches(word, "as")) return AS;
303             break;
304         case 'b':
305             if(matches(word, "break")) return BREAK;
306             break;
307         case 'c':
308             if (matches(word, "call")) return CALL;
309             if (matches(word, "class___")) return CLASS;
310             if (matches(word, "continue")) return CONTINUE;
311             break;
312         case 'd':
313             if (matches(word, "define")) return DEFINE;
314             break;
315         case 'e':
316             if (matches(word, "else")) return ELSE;
317             break;
318         case 'f':
319             if (matches(word, "foreach")) return FOREACH;
320             if (matches(word, "false")) return FALSE;
321             break;
322         case 'i':
323             if (matches(word, "if")) return IF;
324             if (matches(word, "import")) return IMPORT;
325             if (matches(word, "in")) return IN;
326             if (matches(word, "isa")) return ISA;
327             break;
328         case 'n':
329             if (matches(word, "null")) return NULL;
330             if (matches(word, "not")) return NOT;
331             break;
332         case 'o':
333             if (matches(word, "or")) return OR;
334             break;
335         case 'r':
336             if (matches(word, "reverse")) return REVERSE;
337             break;
338         case 't':
339             if (matches(word, "true")) return TRUE;
340             if (matches(word, "template")) return TEMPLATE;
341             break;
342         }
343 
344         return UNKNOWN;
345     }

347     /**
348      * Case sensitive match test.
349      * @param val must be lowercase
350      */
351     private static boolean matches(StringBuilder word, String val) {
352         int len = word.length();
353         if (len != val.length()) return false;
354 
355         // Start at index 1, assuming that the first characters have already
356         // been checked to match.
357         for (int index = 1; index < len; index++) {
358             char cw = word.charAt(index);
359             char cv = val.charAt(index);
360 
361             if (cw != cv) {
362                 return false;
363             }
364         }
365 
366         return true;
367     }

490     public String toString() {
491         StringBuilder buf = new StringBuilder(10);
492 
493         String image = getImage();
494 
495         if (image != null) {
496             buf.append(image);
497         }
498 
499         String str = getStringValue();
500 
501         if (str != null) {
502             if (image != null) {
503                 buf.append(' ');
504             }
505             buf.append(str);
506         }
507 
508         return buf.toString();
509     }
