48  /**
49   *
50   * @author Guy A. Molinari
51   *
52   * The TemplateRepository class manages metadata pertaining to the
53   * caller/callee hierarchy.  This information is derived from the template
54   * class files and does not rely on tea source code.  In order to reduce the
55   * repository spin-up time, this data is stored in a file named <b>.templates.info</b>
56   * and is updated as templates are successfully compiled.  This file is located
57   * in the root directory where template classes are stored.
58   *
59   */
60  public class TemplateRepository {

63      // private File mRepositoryFile;

67          new HashMap<String, TemplateInfo>();

70      private HashMap<String, Map<String, TemplateInfo>> mFunctionMap = null;
71      private static final SimpleDateFormat mDateFmt = new SimpleDateFormat("yyyy-MM-dd hh:mm:ss(S) a z");
72      private static TemplateRepository mInstance = null;

76      private TemplateRepository(File rootClassesDir, String rootPackage) {
77          mRootClassesDir = rootClassesDir;
78          mRootPackage = rootPackage.replace('.', '/');
79          mRootPackage = !mRootPackage.endsWith("/") ? mRootPackage += "/" : mRootPackage;
80      }

82      /**
83       * Initalizes and loads the repository.  Note:  call this method instead
84       * of a constructor to initialize the singleton instance.
85       *
86       * @param rootClassesDir The root template classes directory.
87       * @param rootPackage The root package name (org.teatrove.teaservlet.template).
88       */
89      public static synchronized void init(File rootClassesDir, String rootPackage) {
90          if (rootClassesDir == null || rootPackage == null)
91              return;
92          if (! rootClassesDir.isDirectory())
93              throw new IllegalArgumentException("Root classes parameter not a directory.");
94          if (mInstance != null)
95              throw new RuntimeException("Repository already initialized.");
96          mInstance = new TemplateRepository(rootClassesDir, rootPackage);
97          try {
98              long start = System.currentTimeMillis();
99              mInstance.loadRepositoryFile();
100             Syslog.info("Repository initialized.  Elapsed time " +
101                 (System.currentTimeMillis() - start) + " ms.");
102         }
103         catch (Exception ex) {
104             long start = System.currentTimeMillis();
105             Syslog.info("Generating repository...");
106             try {
107                 mInstance.getTemplateInfoForAllFiles();
108                 mInstance.createRepositoryFile();
109             }
110             catch (IOException ix) {
111                 ix.printStackTrace();
112             }
113             Syslog.info("Repository created.  Elapsed time " +
114                  (((double) (System.currentTimeMillis() - start)) / 1000) + " sec.");
115         }
116     }

136      * Retrieves template metadata for a class file.
137      *
138      * @param f The class file object.
139      * @return TemplateInfo The template metadata.
140      */
141     public TemplateInfo getTemplateInfoForClassFile(File f) throws IOException {
142 
143         if (!f.exists())
144             return null;
145         MethodInfo mi = TemplateCallExtractor.getTemplateExecuteMethod(
146             new FileInputStream(f));
147         return mi != null ? new TemplateInfo(mi, f.lastModified(), mRootClassesDir) : null;

150     /**
151      * Template metadata container object.
152      */
153     public class TemplateInfo {
154         String mName;
155         long mLastModified;
156         TypeDesc mReturnType;
157         TypeDesc[] mParameterTypes;
158         String[] mDependents;
159         TemplateCallExtractor.AppMethodInfo[] mMethodsCalled;
160         boolean mPrecompiled = false;
161 
162         /**
163          * Construct a TemplateInfo from MethodInfo.
164          *
165          * @param mi MethodInfo instance.
166          * @param lastModified Last file modification timestamp.
167          * @param rootClassesDir Root template classes location.
168          * @param precompiled true if pre-compiled.
169          */
170         TemplateInfo(MethodInfo mi, long lastModified, File rootClassesDir, boolean precompiled) {
171             mLastModified = lastModified;
172             mName = mi.getClassFile().getClassName().replace('.', '/');;
173             mReturnType = mi.getMethodDescriptor().getReturnType();
174             mParameterTypes = mi.getMethodDescriptor().getParameterTypes();
175             mDependents = TemplateCallExtractor.getTemplatesCalled(
176                 rootClassesDir.getAbsolutePath(), mi.getClassFile().getClassName());
177             if (mParameterTypes.length > 0) {
178                 mMethodsCalled = TemplateCallExtractor.getAppMethodsCalled(
179                     rootClassesDir.getAbsolutePath(), mi.getClassFile().getClassName(),
180                     mParameterTypes[0].toString());
181             }
182             else
183                 mMethodsCalled = new TemplateCallExtractor.AppMethodInfo[0];
184             mPrecompiled = precompiled;
185         }
186 
187         /**
188          * Construct a TemplateInfo from a PropertyMap extracting a single
189          * name from the map.
190          *
191          * @param name The template name.
192          * @param p The property map.
193          */
194         TemplateInfo(String name, PropertyMap p) {
195             mName = name;
196             try {
197                 mLastModified = mDateFmt.parse((String) p.get("lastModified")).
198                     getTime();
199             }
200             catch (ParseException ignore) { }
201             mReturnType = TypeDesc.forDescriptor((String) p.get("returnType"));
202             ArrayList<TypeDesc> paramList = new ArrayList<TypeDesc>();
203             if (p.get("parameterTypes") != null) {
204                 StringTokenizer params = new StringTokenizer(
205                     (String) p.get("parameterTypes"), ",", false);
206                 while(params.hasMoreTokens())
207                     paramList.add(TypeDesc.forDescriptor(
208                         ((String) params.nextToken()).trim()));
209             }
210             mParameterTypes = (TypeDesc[]) paramList.toArray(
211                 new TypeDesc[paramList.size()]);
212             ArrayList<String> depList = new ArrayList<String>();
213             if (p.get("dependents") != null) {
214                 StringTokenizer deps = new StringTokenizer(
215                     (String) p.get("dependents"), ",", false);
216                 while(deps.hasMoreTokens())
217                     depList.add(deps.nextToken().trim());
218             }
219             mDependents = (String[]) depList.toArray(
220                 new String[depList.size()]);
221             ArrayList<TemplateCallExtractor.AppMethodInfo> methodList =
222                 new ArrayList<TemplateCallExtractor.AppMethodInfo>();
223             if (p.get("methodsCalled") != null) {
224                 StringTokenizer meths = new StringTokenizer(
225                     (String) p.get("methodsCalled"), ",", false);
226                 while(meths.hasMoreTokens())
227                     methodList.add(new TemplateCallExtractor.AppMethodInfo(meths.nextToken().trim()));
228             }
229             mMethodsCalled = (TemplateCallExtractor.AppMethodInfo[])
230                 methodList.toArray(new TemplateCallExtractor.AppMethodInfo[methodList.size()]);
231             mPrecompiled = "true".equals(p.get("precompiled"));
232         }

247         /**
248          * Returns false if the template signatures do not match.  The
249          * template modification date and other info are ignored.
250          */
251         public boolean equals(TemplateInfo t) {
252             if (t == null || !this.getName().equals(t.getName()) ||
253                 !this.getReturnType().equals(t.getReturnType()) ||
254                 this.getParameterTypes().length != t.getParameterTypes().length)
255                 return false;
256 
257             for (int i = 0; i < this.getParameterTypes().length; i++) {
258                 if (!this.getParameterTypes()[i].equals(
259                         t.getParameterTypes()[i]))
260                     return false;
261             }
262             return true;
263         }

287         /**
288          * Returns a String in 'PropertyMap' format.
289          */
290         public String toString() {
291             String indent = "        ";
292             StringBuffer buf = new StringBuffer();
293             buf.append("    \"").append(mName).append("\" {\n");
294             buf.append(indent).append("lastModified = ").append(
295                 mDateFmt.format(new Date(mLastModified))).append("\n");
296             buf.append(indent).append("returnType = ").append(mReturnType).
297                 append("\n");
298             if (mParameterTypes.length > 0) {
299                 buf.append(indent).append("parameterTypes = ");
300                 for (int i = 0; i < mParameterTypes.length; i++) {
301                     buf.append(mParameterTypes[i]);
302                     if (i < mParameterTypes.length - 1)
303                         buf.append(", ");
304                 }
305                 buf.append("\n");
306             }
307             if (mDependents.length > 0) {
308                 buf.append(indent).append("dependents = ");
309                 for (int i = 0; i < mDependents.length; i++) {
310                     buf.append(mDependents[i]);
311                     if (i < mDependents.length - 1)
312                         buf.append(", ");
313                 }
314                 buf.append("\n");
315             }
316             if (mMethodsCalled.length > 0) {
317                 buf.append(indent).append("methodsCalled = ");
318                 for (int i = 0; i < mMethodsCalled.length; i++) {
319                     buf.append(mMethodsCalled[i]);
320                     if (i < mMethodsCalled.length - 1)
321                         buf.append(", ");
322                 }
323                 buf.append("\n");
324             }
325             buf.append(indent).append("precompiled = " + mPrecompiled + "\n");
326             buf.append("    }\n");
327             return buf.toString();
328         }

331     /**
332      * This method is called to begin searching the classes directory
333      * and all subdirectories. Jar files are also processed.
334      */
335     private void getTemplateInfoForAllFiles() throws IOException {
336         getTemplateInfoForAllFiles(mRootClassesDir);
337         getTemplateInfoForPrecompiledTemplates();
338         buildAncestorMap();
339     }

341     /**
342      * Perform a depth first recursion of subdirectories, then process
343      * all class files in the current directory.
344      */
345     private void getTemplateInfoForAllFiles(File startingFileContext) throws IOException {
346 
347         // Do a depth first recursion
348         File[] dirList = startingFileContext.listFiles(
349                 new FilenameFilter() {
350             public boolean accept(File f, String name) {
351                 return f.isDirectory();
352             }
353         });
354 
355         for (int i = 0; dirList != null && i < dirList.length; i++)
356             getTemplateInfoForAllFiles(dirList[i]);
357 
358         // Process class files in current context
359         File[] classList = startingFileContext.listFiles(
360                 new FilenameFilter() {
361             public boolean accept(File f, String name) {
362                 return f.canRead() && name.endsWith(".class");
363             }
364         });
365 
366         for (int i = 0; classList != null && i < classList.length; i++) {
367             TemplateInfo t = getTemplateInfoForClassFile(classList[i]);
368             if (t != null)
369                 mTemplateInfoMap.put(t.getName(), t);
370         }
371     }

373     /**
374      * Rip apart all jar files in the classpath looking for pre-compiled templates.
375      */
376     private void getTemplateInfoForPrecompiledTemplates() {
377 
378         StringTokenizer paths = new StringTokenizer(
379             System.getProperty("java.class.path"), File.pathSeparator, false);
380         while(paths.hasMoreTokens()) {
381             String name = (String) paths.nextToken();
382             if (!name.endsWith(".jar"))
383                 continue;
384             try {
385                 JarFile jar = null;
386                 try {
387                     jar = new JarFile(name);
388                 }
389                 catch (ZipException zx) {
390                     Syslog.warn("The jar file " + name + " is in the classpath but does not exist or is unreadable.");
391                     continue;
392                 }
393                 for (Enumeration<JarEntry> e = jar.entries(); e.hasMoreElements(); ) {
394                     JarEntry entry = e.nextElement();
395                     if (! entry.getName().startsWith(TemplateCallExtractor.TEMPLATE_PACKAGE.replace('.', '/')) ||
396                             ! entry.getName().endsWith(".class"))
397                         continue;
398 
399                     MethodInfo mi = TemplateCallExtractor.getTemplateExecuteMethod(jar.getInputStream(entry));
400                     if (mi != null) {
401                         TemplateInfo t =  new TemplateInfo(mi, entry.getTime(), mRootClassesDir, true);
402                         mTemplateInfoMap.put(t.getName(), t);
403                     }
404                 }
405             }
406             catch (JarException jx) {
407                 Syslog.error(jx);
408             }
409             catch (IOException ix) {
410                 Syslog.error(ix);
411             }
412         }
413

419     /**
420      * Persist the current state of the repository to disk.
421      */
422     public synchronized void createRepositoryFile() throws IOException {
423         StringBuffer buf = new StringBuffer();
424         long lastModified = System.currentTimeMillis();
425         buf.append("lastModified = ").append(
426             mDateFmt.format(new Date(lastModified))).append("\n");
427         buf.append("templates {\n");
428         for (Iterator<String> i = mTemplateInfoMap.keySet().iterator(); i.hasNext(); )
429             buf.append(((TemplateInfo) mTemplateInfoMap.get(
430                 i.next())).toString());
431         buf.append("}\n");
432         File repositoryFile = new File(mRootClassesDir,
433             REPOSITORY_FILENAME);
434         FileOutputStream out = new FileOutputStream(repositoryFile);
435         out.write(buf.toString().getBytes());
436         out.flush();
437         out.close();
438         repositoryFile.setLastModified(lastModified);
439     }

441     /**
442      * Loads the current repository state from disk.
443      */
444     private void loadRepositoryFile() throws IOException, ParseException {
445         PropertyMap m = new PropertyMap();
446         PropertyParser p = new PropertyParser(m);
447         File repositoryFile = new File(mRootClassesDir,
448             REPOSITORY_FILENAME);
449         p.parse(new FileReader(repositoryFile));
450         boolean isWindows = System.getProperty("os.name").startsWith("Windows");
451         long lastModified = mDateFmt.parse((String) m.get("lastModified")).
452             getTime();
453         if ((isWindows && repositoryFile.lastModified() != lastModified) ||
454                 (!isWindows && Math.max(repositoryFile.lastModified(), lastModified) -
455                 Math.min(repositoryFile.lastModified(), lastModified) >= 1000L)) {
456             String corruptMsg = "Repository corrupt.  Rebuild Needed.";
457             Syslog.error(corruptMsg);
458             throw new RuntimeException(corruptMsg);
459         }
460         PropertyMap templateMap = m.subMap("templates");
461 
462         // Merge the TemplateInfos into existing mTemplateInfoMap
463         for (Iterator<?> i = templateMap.subMapKeySet().iterator(); i.hasNext();) {
464             String key = (String) i.next();
465             mTemplateInfoMap.put(key, new TemplateInfo(key,
466                 templateMap.subMap(key)));
467         }
468         buildAncestorMap();
469     }

471     /**
472      * Create caller hierarchy.
473      *
474      * JoshY 12/28/04 - Due to a bug in the 1.4.2_05 server VM, it's important that you not
475      * use the TemplateCallExtractor.AppMethodInfo as a map key.  We're using a String
476      * representation here as a workaround.
477      */
478     private void buildAncestorMap() {
479         mAncestorMap = new HashMap<String, Map<String, TemplateInfo>>();
480         mFunctionMap = new HashMap<String, Map<String, TemplateInfo>>();
481         for (Iterator<TemplateInfo> i = mTemplateInfoMap.values().iterator(); i.hasNext();) {
482             TemplateInfo t = i.next();
483             for (int j = 0; j < t.getDependents().length; j++) {
484                 Map<String, TemplateInfo> parentMap =
485                     mAncestorMap.get(t.getDependents()[j]);
486                 if (parentMap == null)
487                     parentMap = new HashMap<String, TemplateInfo>();
488                 if (!parentMap.containsKey(t.getName()))
489                     parentMap.put(t.getName(), t);
490                 mAncestorMap.put(t.getDependents()[j], parentMap);
491             }
492             for (int j = 0; j < t.getAppMethodsCalled().length; j++) {
493                 String sKey = t.getAppMethodsCalled()[j].toString();
494                 Map<String, TemplateInfo> callMap = mFunctionMap.get(sKey);
495                 if (callMap == null) {
496                     callMap = new HashMap<String, TemplateInfo>();
497                     mFunctionMap.put(sKey, callMap);
498                 }
499                 if (!callMap.containsKey(t.getName()))
500                     callMap.put(t.getName(), t);
501             }
502         }
503     }

505     /**
506      * Retrieve the metadata for all templates that call the named
507      * template.  The template name can be the short form, long form
508      * and use either '.' or '/' separators.
509      *
510      * @param templateName The name of the template.
511      * @return TemplateInfo[] The callers (consumers).
512      */
513     public TemplateInfo[] getCallers(String templateName) {
514         templateName = !templateName.startsWith(mRootPackage) ?
515             getFullyQualifiedTemplateName(templateName) : templateName;
516         if (mAncestorMap.containsKey(templateName)) {
517             Collection<TemplateInfo> parents = mAncestorMap.get(templateName).values();
518             return parents.toArray(new TemplateInfo[parents.size()]);
519         }
520         else
521             return new TemplateInfo[0];
522     }

524     /**
525      * Retrieve the metadata for all templates that call the specified
526      * method.
527      *
528      * @param methodDesc The method descriptor to find.
529      * @return TemplateInfo[] The callers (consumers).
530      */
531     public TemplateInfo[] getMethodCallers(MethodDescriptor methodDesc) {
532         Method method = methodDesc.getMethod();
533         Class<?>[] paramClasses = method.getParameterTypes();
534         java.lang.reflect.Type[] paramTypes = method.getGenericParameterTypes();
535 
536         TypeDesc[] params = new TypeDesc[paramClasses.length];
537         for (int i = 0; i < params.length; i++)
538             params[i] = TypeDesc.forClass(paramClasses[i], paramTypes[i]);
539         TemplateCallExtractor.AppMethodInfo mi =
540             new TemplateCallExtractor.AppMethodInfo(method.getName(), params);
541 
542         if (mFunctionMap.containsKey(mi)) {
543             Collection<TemplateInfo> callers = mFunctionMap.get(mi).values();
544             return callers.toArray(new TemplateInfo[callers.size()]);
545         }
546         else
547             return new TemplateInfo[0];
548     }

577     /**
578      * Retrieve the callers of a list of given template names
579      * where the source file can be located (delegated to the compiler),
580      * and the signature of given template(s) has changed since the last
581      * call to <b>update</b>.  The class file(s) of the corresponding
582      * template names returned are removed.
583      *
584      * @param names The names of the templates to check.
585      * @param compiler The compiler instance.
586      * @return String[] The caller templates needing recompilation.
587      */
588     public String[] getCallersNeedingRecompile(String[] names, Compiler compiler) throws IOException {
589         HashMap<String, TemplateInfo> needsCompile =
590             new HashMap<String, TemplateInfo>();
591 
592         for (int i = 0; i < names.length; i++) {
593             String templateName = getFullyQualifiedTemplateName(names[i]);
594             TemplateInfo tNew = getTemplateInfoForClassFile(getClassFileForName(templateName));
595             if (tNew == null || !mTemplateInfoMap.containsKey(templateName))
596                 continue;
597             TemplateInfo tOld = (TemplateInfo) mTemplateInfoMap.get(templateName);
598             // Don't bother if template signature hasn't changed.
599             if (tNew.equals(tOld))
600                 continue;
601             TemplateInfo[] callers = getCallers(templateName);
602             for (int j = 0; j < callers.length; j++) {
603                 String shortPath = callers[j].getName().
604                     substring(mRootPackage.length()).replace('/','.');
605                 if (!needsCompile.containsKey(shortPath) &&
606                         compiler.sourceExists(shortPath)) {
607                     needsCompile.put(shortPath, callers[j]);
608                     // Nuke the caller's class file to force a recompile.
609                     getClassFileForName(callers[j].getName()).delete();
610                 }
611             }
612         }
613         return needsCompile.keySet().toArray(new String[needsCompile.size()]);
614     }

616     /**
617      * Synchronize the repository state with respect to the named template
618      * class file(s).  The dependency lists will be updated an the new
619      * repository file will be written to disk.
620      *
621      * @param templatesChanged The templates to update.
622      */
623     public synchronized void update(String[] templatesChanged) throws IOException {
624         HashMap<String, String> updated = new HashMap<String, String>();
625         for (int i = 0; i < templatesChanged.length; i++) {
626             String templateName = getFullyQualifiedTemplateName(templatesChanged[i]);
627             if (!updated.containsKey(templateName)) {
628                 TemplateInfo tUpdated = getTemplateInfoForClassFile(getClassFileForName(templateName));
629                 if (tUpdated != null)
630                     mTemplateInfoMap.put(templateName, tUpdated);
631                 else
632                     mTemplateInfoMap.remove(templateName);
633                 updated.put(templateName, templateName);
634             }
635         }
636         buildAncestorMap();
637         createRepositoryFile();
638     }

640     /**
641      * Canonicalize the template name.
642      */
643     private String getFullyQualifiedTemplateName(String name) {
644         String s = name.replace('.', '/');
645         return !s.startsWith(mRootPackage) ? mRootPackage + s : s;
646     }

648     /**
649      * Get the class file for a template name.
650      */
651     private File getClassFileForName(String name) throws IOException {
652         File templateFile = null;
653         if (name.startsWith(mRootPackage))
654             templateFile = new File(mRootClassesDir,
655                 name.substring(mRootPackage.length()) + ".class");
656 
657         if (templateFile == null || !templateFile.exists())
658             templateFile = new File(mRootClassesDir, name + ".class");
659 
660         return templateFile;
661     }
