17  package org.teatrove.tea.compiler;
18  
19  import java.beans.IntrospectionException;
20  import java.lang.reflect.Method;
21  
22  /**
23   * This class finds methods that best fit a given description. The compiler
24   * will then bind to one of those methods.
25   *
26   * @author Brian S O'Neill
27   */
28  public class MethodMatcher {
29      /**
30       * The best result candidates are stored in the Method array passed in.
31       * The int returned indicates the number of candidates in the array. Zero
32       * is returned if there is no possible match.
33       */
34      public static int match(Method[] methods, String name, Type... params) {
35          int paramCount = params.length;
36          int matchCount = methods.length;
37          Method m;
38  
39          int[] costs = new int[matchCount];
40  
41          // Filter the available methods down to a smaller set, tossing
42          // out candidates that could not possibly match because the name
43          // differs, and the number of parameters differ. Also eliminate
44          // ones in which the parameter types are not compatible at all
45          // because no known conversion could be applied.
46  
47          int lowestTotalCost = Integer.MAX_VALUE;
48          int length = matchCount;
49          matchCount = 0;
50          for (int i=0; i < length; i++) {
51              m = methods[i];
52              if (name == null || m.getName().equals(name)) {
53                  Class<?>[] methodParams = m.getParameterTypes();
54                  
55                  int cnt = methodParams.length;
56                  if ((cnt == paramCount) || 
57                      (m.isVarArgs() && paramCount >= cnt - 1)) {
58                      
59                      int j = 0;
60                      int total = 0;
61                      for (; j < paramCount; j++) {
62                          Type methodParam = getMethodParam(m, j, params[j]);
63                          int cost = methodParam.convertableFrom(params[j]);
64                          
65                          if (cost < 0) { break; }
66                          else { total += cost; }
67                      }
68  
69                      if (j == paramCount) {
70                          // discount non-use of var args
71                          if (m.isVarArgs() && j < methodParams.length) {
72                              total++;
73                          }
74                          
75                          costs[matchCount] = total;
76                          methods[matchCount++] = m;
77                          if (total < lowestTotalCost) {
78                              lowestTotalCost = total;
79                          }
80                      }
81                  }
82              }
83          }
84  
85          if (matchCount <= 1) {
86              return matchCount;
87          }
88  
89          // Filter out those that have a cost higher than lowestTotalCost.
90          length = matchCount;
91          matchCount = 0;
92          for (int i=0; i < length; i++) {
93              if (costs[i] <= lowestTotalCost) {
94                  costs[matchCount] = costs[i];
95                  methods[matchCount++] = methods[i];
96              }
97          }
98  
99          if (matchCount <= 1) {
100             return matchCount;
101         }
102 
103         // Filter further by matching parameters with the shortest distance
104         // in the hierarchy. This matches each parameter to each matching method
105         // to either find the best possible match where the method parameter is
106         // the shortest. If multiple matches exist, the parameter is considered
107         // ambiguous. If all parameters are ambiguous, then this will return -1
108         // as a failiure. If multiple methods match for different parmaters,
109         // then the result is also considered ambiguous and -1 returned. For
110         // example, the methods:
111         //     void doX(String a, Object b)
112         //     void doX(Object a, String b)
113         // with the method: 
114         //     doX("string1", "string2")
115         // will best match the first method as String is closer than object but
116         // will best match the second method as again String is closer. Thus,
117         // the result is ambiguous.
118 
119         // maintain the best matching method per parameter
120         Method[] paramResults = new Method[paramCount];
121         
122         // search each parameter separately to find best matching method
123         for (int j=0; j<paramCount; j++) {
124             Class<?> lastMatch = null;
125             Method bestFit = null;
126             
127             length = matchCount;
128             int bestFits = 0;
129             
130             // search each matching method for the best match. If we have not
131             // yet marked a best fit, then we mark the first method. Otherwise,
132             // if the next method is closer (ie: the last match is assignable)
133             // then we update the best fit. If the last match ever matches
134             // the current match, then it is considered ambiguous and the best
135             // fit is reset.
136             
137             for (int i=0; i < length; i++) {
138                 m = methods[i];
139                 
140                 Type methodType = getMethodParam(m, j, params[j]);
141                 Class<?> methodParam = methodType.getNaturalClass();
142                 
143                 // TODO: match against generics?
144                 //java.lang.reflect.Type methodType = methodType.getGeneric();
145 
146                 Class<?> param = params[j].getNaturalClass();
147                 if (methodParam.isAssignableFrom(param)) {
148                 	
149                 	// check for ambiguity by determining if multiple methods
150                 	// have same type
151                 	
152                     if (lastMatch != null) {
153                         if (lastMatch.equals(methodParam)) {
154                             bestFits++;
155                         }
156                         else { bestFits = 0; }
157                     }
158 
159                     // if first time through or last match is assignable (ie:
160                     // farther away), then update best fit
161                     
162                     if (lastMatch == null ||
163                         lastMatch.isAssignableFrom(methodParam)) {
164                         
165                         bestFit = m;
166                         lastMatch = methodParam;
167                     }
168                 }
169             }
170 
171             // set best fitting method if single match...otherwise, leave null
172             // to denote ambiguous result
173             
174             if (bestFits == 0) {
175             	paramResults[j] = bestFit;
176             }
177         }
178         
179         // check if multiple matching methods (ambiguous) or if all params are
180         // ambiguous
181         
182         Method last = null;
183         for (int i = 0; i < paramCount; i++) {
184         	// update last match if not yet determined
185         	if (last == null) {
186         		last = paramResults[i];
187         	}
188         	
189     		// check if non-match (ie: multiple matching methods), and return
190         	// ambiguous (-1)
191         	else if (!last.equals(paramResults[i])) {
192         		return -1;
193         	}
194         }
195         
196         // check if all ambiguous..no matches
197         if (last == null) {
198         	return -1;
199         }
200         
201         // ensure first result is set to match
202         methods[0] = last;
203         
204         // return single result
205         return 1;
206     }

208     public static Type getMethodParam(Method method, int index, Type type) {
209         Type result = null;
210         Class<?>[] methodParams = method.getParameterTypes();
211         java.lang.reflect.Type[] methodTypes = method.getGenericParameterTypes();
212         
213         if (method.isVarArgs() && index >= methodParams.length - 1) {
214             int idx = methodParams.length - 1;
215             Type varArg = new Type(methodParams[idx], methodTypes[idx]);
216             if (index == methodParams.length - 1) {
217                 if (varArg.convertableFrom(type) >= 0) {
218                     result = varArg;
219                 }
220             }
221             
222             if (result == null) {
223                 try { result = varArg.getArrayElementType(); }
224                 catch (IntrospectionException ie) {
225                     throw new RuntimeException(ie);
226                 }
227             }
228             else {
229                 
230             }
231         }
232         else if (index < methodParams.length) {
233             result = new Type(methodParams[index], methodTypes[index]);
234         }
235         
236         return result;
237     }
