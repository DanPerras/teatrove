17  package org.teatrove.tea.engine;
18  
19  /**
20   * Instances of this kind of ContextSource can support reloading their context
21   * class. {@link TeaExecutionEngine Engines} that support context reload will
22   * discover that the context class has changed by calling getContextType. This
23   * call will likely be initiated by a template reload request, since all
24   * templates may need to be recompiled if the context class has changed.
25   * <p>
26   * Because Tea engines may execute many templates concurrently, the two
27   * argument createContext method is called to inform the context source
28   * what specific context class is tied to the application request. The returned
29   * context instance must be of the requested type. DynamicContextSources must
30   * support instantiating the old context class(es) as well as the new context
31   * class.
32   * <p>
33   * The context source may release old context classes to free resources only
34   * when it is safe to do so. One way of achieving this is by using weak
35   * references.
36   *
37   * @author Jonathan Colwell
38   */
39  public interface DynamicContextSource extends ContextSource {
40  
41      /**
42       * Creates a context instance that must be of the requested type.
43       *
44       * @param contextClass Expected type of returned context instance
45       * @param param {@link TeaExecutionEngine engine} specific parameter that
46       * may be used to create the context instance
47       */
48      public Object createContext(Class<?> contextClass, Object param)
49          throws Exception;
50  }
