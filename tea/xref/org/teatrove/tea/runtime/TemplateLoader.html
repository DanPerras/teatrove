27  /**
28   * TemplateLoader manages the loading and execution of Tea templates. To
29   * reload templates, create a new TemplateLoader with a new ClassLoader.
30   *
31   * @author Brian S O'Neill
32   * @see org.teatrove.tea.util.ClassInjector
33   */
34  public class TemplateLoader {
35      private ClassLoader mBaseLoader;
36      private String mPackagePrefix;
37  
38      // Maps full template names to Templates.
39      private Map<String, Template> mTemplates;
40  
41      /**
42       * Creates a TemplateLoader that uses the current ClassLoader as a base.
43       * It is recommended that templates be compiled to a base package and that
44       * a TemplateLoader be constructed with the package prefix. This way,
45       * the TemplateLoader can easily distinguish between template classes and
46       * normal classes, only loading the templates.
47       */
48      public TemplateLoader() {
49          this(null);
50      }
51  
52      /**
53       * Creates a TemplateLoader that uses the current ClassLoader as a base.
54       *
55       * @param packagePrefix Package that templates should be loaded from
56       */
57      public TemplateLoader(String packagePrefix) {
58          init(getClass().getClassLoader(), packagePrefix);
59      }
60  
61      /**
62       * Creates a TemplateLoader that uses the given ClassLoader as a base. A
63       * base ClassLoader is used to load both template and non-template classes.
64       * The base ClassLoader can use the package prefix for determining
65       * whether or not it is loading a template.
66       *
67       * @param baseLoader Base ClassLoader
68       * @param packagePrefix Package that templates should be loaded from
69       */
70      public TemplateLoader(ClassLoader baseLoader, String packagePrefix) {
71          init(baseLoader, packagePrefix);
72      }
73  
74      private void init(ClassLoader baseLoader, String packagePrefix) {
75          mBaseLoader = baseLoader;
76  
77          if (packagePrefix == null) {
78              packagePrefix = "";
79          }
80          else if (!packagePrefix.endsWith(".")) {
81              packagePrefix += '.';
82          }
83          mPackagePrefix = packagePrefix.trim();
84  
85          mTemplates = new HashMap<String, Template>();
86      }
87  
88      /**
89       * Get or load a template by its full name. The full name of a template
90       * has '.' characters to separate name parts, and it does not include a
91       * Java package prefix.
92       *
93       * @throws ClassNotFoundException when template not found
94       * @throws NoSuchMethodException when the template is invalid
95       */
96      public final synchronized Template getTemplate(String name)
97          throws ClassNotFoundException, NoSuchMethodException, LinkageError
98      {
99          Template template = (Template)mTemplates.get(name);
100         if (template == null) {
101             template = loadTemplate(name);
102             mTemplates.put(name, template);
103         }
104         return template;
105     }
106 
107     /**
108      * Returns all the templates that have been loaded thus far.
109      */
110     public final synchronized Template[] getLoadedTemplates() {
111         return (Template[])mTemplates.values().toArray
112             (new Template[mTemplates.size()]);
113     }
114 
115     protected Template loadTemplate(String name)
116         throws ClassNotFoundException, NoSuchMethodException, LinkageError
117     {
118         return new TemplateImpl
119             (name, mBaseLoader.loadClass(mPackagePrefix + name));
120     }

148         /**

153         public String[] getParameterNames();

179     private class TemplateImpl implements Template {
180         private String mName;
181         private Class<?> mClass;
182 
183         private transient Method mExecuteMethod;
184         private transient Class<?> mReturnType;
185         private transient String[] mParameterNames;
186         private transient Class<?>[] mParameterTypes;
187 
188         private TemplateImpl(String name, Class<?> clazz)
189             throws NoSuchMethodException
190         {
191             mName = name;
192             mClass = clazz;
193             doReflection();
194         }
195 
196         public TemplateLoader getTemplateLoader() {
197             return TemplateLoader.this;
198         }
199 
200         public String getName() {
201             return mName;
202         }
203 
204         public Class<?> getTemplateClass() {
205             return mClass;
206         }
207 
208         public Class<?> getContextType() {
209             return mExecuteMethod.getParameterTypes()[0];
210         }
211 
212         public String[] getParameterNames() {
213             return (String[])mParameterNames.clone();
214         }
215 
216         public Class<?>[] getParameterTypes() {
217             return (Class[])mParameterTypes.clone();
218         }
219 
220         public void execute(Context context, Object[] parameters)
221             throws Exception
222         {
223             int length = parameters.length;
224             Object[] args = new Object[1 + length];
225             args[0] = context;
226             for (int i=0; i<length; i++) {
227                 args[i + 1] = parameters[i];
228             }
229 
230             try {
231                 Object ret = mExecuteMethod.invoke(null, args);
232                 if (mReturnType != void.class) {
233                     context.print(ret);
234                 }
235             }
236             catch (InvocationTargetException e) {
237                 Throwable t = e.getTargetException();
238                 if (t instanceof Exception) {
239                     throw (Exception)t;
240                 }
241                 else if (t instanceof Error) {
242                     throw (Error)t;
243                 }
244                 else {
245                     throw e;
246                 }
247             }
248         }
249 
250         public String toString() {
251             StringBuffer buf = new StringBuffer(80);
252 
253             buf.append("template ");
254             buf.append(getName());
255             buf.append('(');
256 
257             buf.append(getContextType().getName());
258 
259             String[] paramNames = getParameterNames();
260             Class<?>[] paramTypes = getParameterTypes();
261             int length = paramTypes.length;
262             for (int i=0; i<length; i++) {
263                 buf.append(", ");
264                 buf.append(paramTypes[i].getName());
265                 if (paramNames[i] != null) {
266                     buf.append(' ');
267                     buf.append(paramNames[i]);
268                 }
269             }
270 
271             buf.append(')');
272 
273             return buf.toString();
274         }

276         private void doReflection() throws NoSuchMethodException {
277             // Bind to first execute method found; there should be one.
278             Method[] methods = getTemplateClass().getMethods();
279 
280             for (int i=0; i<methods.length; i++) {
281                 Method m = methods[i];
282                 if (m.getName().equals
283                     (JavaClassGenerator.EXECUTE_METHOD_NAME) &&
284                     Modifier.isStatic(m.getModifiers())) {
285 
286                     mExecuteMethod = m;
287                     break;
288                 }
289             }
290 
291             if (mExecuteMethod == null) {
292                 throw new NoSuchMethodException
293                     ("No execute method found in class " +
294                      "for template \"" + getName() + "\"");
295             }
296 
297             mReturnType = mExecuteMethod.getReturnType();
298 
299             Class<?>[] methodParams = mExecuteMethod.getParameterTypes();
300             if (methodParams.length == 0 ||
301                 !Context.class.isAssignableFrom(methodParams[0])) {
302 
303                 throw new NoSuchMethodException
304                     ("Execute method does not accept a context " +
305                      "for template \"" + getName() + "\"");
306             }
307 
308             int length = methodParams.length - 1;
309             mParameterNames = new String[length];
310             mParameterTypes = new Class[length];
311 
312             for (int i=0; i<length; i++) {
313                 mParameterTypes[i] = methodParams[i + 1];
314             }
315 
316             try {
317                 Method namesMethod = getTemplateClass().getMethod
318                     (JavaClassGenerator.PARAMETER_METHOD_NAME);
319 
320                 String[] names = (String[])namesMethod.invoke(null);
321                 if (names != null) {
322                     // Copy, just in case the length differs.
323                     for (int i=0; i<length; i++) {
324                         mParameterNames[i] = names[i];
325                     }
326                 }
327             }
328             catch (Exception e) {
329                 // No big deal, we just don't set paramater names.
330             }
331         }
