30       */
31      public File[] getFiles() {
32  
33          String path = mRequest.getParameter("path");
34          if (path == null) {
35              path = mApp.getInitParameter("defaultPath");
36          }
37          if (path == null) {
38              path = "/";
39          }
40  
41          File activefile = new File(path);
42          if( activefile.isDirectory()) {
43              return activefile.listFiles();
44          } else {
45              return null;
46          }
