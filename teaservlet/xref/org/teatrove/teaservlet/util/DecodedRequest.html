17  package org.teatrove.teaservlet.util;
18  
19  import java.io.UnsupportedEncodingException;
20  import java.util.Set;
21  import java.util.HashSet;
22  import javax.servlet.http.HttpServletRequest;
23  import javax.servlet.http.HttpServletRequestWrapper;
24  
25  /**
26   * A convenience HttpServletRequest wrapper that automatically decodes request
27   * parameters using the provided character encoding.
28   *
29   * @author Brian S O'Neill
30   */
31  public class DecodedRequest extends HttpServletRequestWrapper {
32      private static final byte[] TEST_BYTES = {65};
33  
34      private static Set cGoodEncodings = new HashSet(7);
35  
36      private static synchronized String checkEncoding(String encoding) {
37          if (!cGoodEncodings.contains(encoding)) {
38              // Test the encoding.
39              try {
40                  new String(TEST_BYTES, encoding);
41              }
42              catch (UnsupportedEncodingException e) {
43                  throw new IllegalArgumentException
44                      ("Unsupported character encoding: " + encoding);
45              }
46              cGoodEncodings.add(encoding);
47          }
48          return encoding;
49      }
50  
51      private final String mEncoding;
52      private final String mOriginalEncoding;
53  
54      /**
55       * @param request wrapped request
56       * @param encoding character encoding to apply to request parameters
57       * @throws IllegalArgumentException when the encoding isn't supported
58       */
59      public DecodedRequest(HttpServletRequest request, String encoding) {
60          super(request);
61          mEncoding = checkEncoding(encoding);
62          String original = request.getCharacterEncoding();
63          if (original == null) {
64              original = "iso-8859-1";
65          }
66          mOriginalEncoding = original;
67      }
68  
69      public String getCharacterEncoding() {
70          return mEncoding;
71      }
72  
73      public String getParameter(String name) {
74          String value;
75          if ((value = super.getParameter(name)) != null) {
76              try {
77                  return new String
78                      (value.getBytes(mOriginalEncoding), mEncoding);
79              }
80              catch (UnsupportedEncodingException e) {
81              }
82          }
83          return value;
84      }
85  
86      public String[] getParameterValues(String name) {
87          String[] values = (String[])super.getParameterValues(name).clone();
88          try {
89              String enc = mEncoding;
90              String orig = mOriginalEncoding;
91              for (int i = values.length; --i >= 0; ) {
92                  String value;
93                  if ((value = values[i]) != null) {
94                      values[i] = new String(value.getBytes(orig), enc);
95                  }
96              }
97          }
98          catch (UnsupportedEncodingException e) {
99          }
100         return values;
101     }
