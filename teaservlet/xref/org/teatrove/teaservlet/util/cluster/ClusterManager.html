73      public static ClusterManager createClusterManager(PropertyMap properties, 
74                                                        Clustered clusterObj) 
75          throws Exception {
76  
77          ClusterManager manager = null;
78  
79          if (properties != null) {
80              properties = properties.subMap("cluster");
81  
82              if (properties.keySet().size() > 0) { 
83                  String servers = properties
84                      .getString("servers");
85                  String clusterName = properties
86                      .getString("name");
87                  InetAddress multicastGroup = null;
88                  String netInterface = properties
89                      .getString("localNet");
90  
91                  int rmiPort = properties.getInt("rmi.port", 1099);
92                      
93                  int multicastPort = properties.getInt("multicast.port", 1099);
94                  String group = properties.getString("multicast.group");
95                  if (group != null) {
96                      try {
97                          multicastGroup = InetAddress.getByName(group);
98                      }
99                      catch (UnknownHostException uhe) {}
100                 }
101                 
102                 HOSTMAP_TIMEOUT = 
103                     properties.getInteger("hostMapTimeout", 
104                                           new Integer(300000)).
105                     longValue();
106 
107                 if (multicastGroup != null) {
108                     manager = new ClusterManager(clusterObj,
109                                                  multicastGroup,
110                                                  multicastPort,
111                                                  rmiPort,
112                                                  netInterface,
113                                                  servers);
114 
115                     manager.setMulticastTtl(properties.getInt("multicast.ttl", 1));
116                 }
117                 else if (servers != null) {
118                     manager = new ClusterManager(clusterObj,
119                                                  rmiPort,
120                                                  netInterface,
121                                                  servers);
122                 }
123             }
124         }
125         return manager;

129      * @param host The machine name whose IP we want to find.
130      * @return the IP Address for the machine specified by <b>host</b>.
131      */
132     private static String nslookup(String host) {
133 
134         InetAddress address = null;
135         String ipAddr = new String();
136 
137         try {
138           address = InetAddress.getByName(host);
139         } catch(UnknownHostException e) {
140             Syslog.error("Unable to find IP for host: " + host);
141             return new String();
142         }
143 
144         byte[] ip = address.getAddress();
145         for (int i=0; i<ip.length; i++) {
146             ipAddr += ((ip[i]) & 0xff);
147             if (i + 1 < ip.length) {
148                 ipAddr += ".";
149             }
150         }
151         return ipAddr;
152     }

244         mExplicitlySpecifiedServerNames = new Vector();
245         if (serverNames != null) {

258     public String[] resolveServerNames() {
259 
260         // return the cached ips for the machines unless the time has
261         // expired or the hostMap is empty.
262         if (!mHostMap.isEmpty() && !hasHostMapExpired()) {
263             // return (String[])mHostMap.values().toArray(new String[0]);
264             return (String[])mHostMap.keySet().toArray(new String[0]);
265         }
266 
267         Map lastKnownGood = new HashMap(mHostMap);
268         mHostMap.clear();
269 
270         /* check if any unresolved servers have come online */
271         synchronized (mUnresolvedServerNames) {
272             
273             Iterator resolveIt = mUnresolvedServerNames.iterator();
274             while (resolveIt.hasNext()) {
275                 String nextHost = (String)resolveIt.next();
276                 try {
277                     String nextIP = new String();
278                     nextIP = nslookup(nextHost);
279                     if (nextIP.length() == 0) {
280                         // Couldn't find the IP. Using the last known 
281                         // good settings.
282                         Syslog.info("Using last known good IP for host: " + 
283                                     nextHost);
284                         nextIP = (String)lastKnownGood.get(nextHost);
285                         if (nextIP == null || nextIP.length() == 0) {
286                             // There is no last known good IP, so just use the 
287                             // host name
288                             Syslog.info("Last known good IP not found. Using the host name");
289                             nextIP = nextHost;
290                         }
291                     }
292 
293                     mHostMap.put(nextHost, nextIP);
294 
295                     String namingURL =
296                         ("//" + nextIP + ':' + mRmiPort
297                          + '/' + mCluster.getClusterName());
298 
299                     if (DEBUG) {
300                         Syslog.debug("Looking Up " + namingURL);
301                     }
302 
303                     Clustered bcl = 
304                         (Clustered)Naming.lookup(namingURL);
305                                     
306                     if (bcl != null) {
307                         if (!mCluster.containsPeer(bcl)) {
308                             resolveIt.remove();
309                             mCluster.addPeer(bcl);
310                             Syslog.info("Successfullly resolved " 
311                                         + bcl.getServerName() +
312                                         " as a member of the " 
313                                         + mCluster.getClusterName() 
314                                         + " cluster.");
315                         }
316                     }
317                     continue;
318                 }           
319                 catch (Exception e) {
320                     // Syslog.warn(e);
321                 }
322                 if (DEBUG) {
323                     Syslog.debug("Failed to resolve " 
324                                  + nextHost + " as part of this cluster");
325                 }
326             }
327         }
328 
329         // set the update time
330         mHostListUpdateTime = System.currentTimeMillis();
331 
332         try {
333             Syslog.debug("Current members of the " 
334                     + mCluster.getClusterName() +
335                     " cluster");       
336 
337             /* check if any servers disappeared */
338             Clustered[] peers = getCluster().getKnownPeers();
339             ArrayList peerNames = new ArrayList();
340             boolean lostOne = false;
341             for (int j = 0;j<peers.length;j++) {
342                 try {
343                     String name = peers[j].getServerName();
344                     Syslog.debug(name);
345                     peerNames.add(name);
346                 }
347                 catch (RemoteException re) {
348                     getCluster().removePeer(peers[j]);
349                 }
350             }
351 
352             mUnresolvedServerNames = 
353                 (Vector)mExplicitlySpecifiedServerNames.clone();
354             mUnresolvedServerNames.removeAll(peerNames);
355             
356             if (mUnresolvedServerNames.size() > 0) {
357                 Syslog.debug("Unresolved server names: ");
358                 Iterator unIt = mUnresolvedServerNames.iterator();
359                 while (unIt.hasNext()) {
360                     Syslog.debug((String)unIt.next());
361                 }
362             }
363             return (String[])peerNames.toArray(new String[peerNames.size()]);
364         }
365         catch (RemoteException re2) {
366             Syslog.warn(re2);
367         }
368 
369         return null;
370     }

509     protected Registry prepareRegistry(Clustered cluster,int port,
510                                        String netInterface)
511         throws IOException {
512 
513         if (mRMISocketFactory == null) {
514             try {
515                 RMISocketFactory factory = RMISocketFactory.getSocketFactory();
516                 if (factory == null) {
517                     factory = new BackchannelSocketFactory
518                         (getHostName(), netInterface);
519                     
520                     if (factory != null) {
521                         RMISocketFactory.setSocketFactory(factory);
522                         Syslog.info("RMI will now operate on " 
523                                     + ((BackchannelSocketFactory)factory)
524                                     .getInetAddress()
525                                     .getHostAddress());
526                     }
527                 }
528                 mRMISocketFactory = factory;
529             }
530             catch (IOException ioe) {
531                 Syslog.warn(ioe);
532             }
533         }
534             
535         Registry reg = null;
536         if (DEBUG) {
537             Syslog.debug("preparing registry on port " + port);
538             Syslog.debug("old port " + cActiveRegistryPort);
539         }
540 
541         if (cActiveRegistryPort < 0) {
542             try {
543                 if (mRMISocketFactory != null) {
544                     reg = LocateRegistry.createRegistry(port, 
545                                                         mRMISocketFactory,
546                                                         mRMISocketFactory);
547                     Syslog.info("RMI registry created on " 
548                                 + ((BackchannelSocketFactory)
549                                    mRMISocketFactory).getInetAddress()
550                                 .getHostAddress()+ ':' + port);                
551                 }
552                 else {
553                     reg = LocateRegistry.createRegistry(port);
554                 }
555                 cActiveRegistryPort = port;        
556             }
557             catch (RemoteException re) {
558                 Syslog.warn(re);
559                 reg = null;     
560             }
561         }
562 
563         if (reg == null) {
564             try {
565                 if (mRMISocketFactory != null) {
566                     reg = LocateRegistry.getRegistry(getHostName(), port, 
567                                                      mRMISocketFactory);
568                     Syslog.info("RMI registry reference created for " 
569                                 + ((BackchannelSocketFactory)
570                                    mRMISocketFactory).getInetAddress()
571                                 .getHostAddress() + ':' + port);
572                 }
573                 else {
574                     reg = LocateRegistry.getRegistry(getHostName(), port);
575                 }
576                 reg.list();
577             }
578             catch (RemoteException re) {    
579                 Syslog.warn(re);
580                 reg = null;     
581             }
582         }
583 
584         if (reg != null) {
585             try {
586                 reg.bind(cluster.getClusterName(),cluster);
587             }
588             catch (AlreadyBoundException abe) {
589                 reg.rebind(cluster.getClusterName(),cluster);
590             }
591             catch (NoSuchObjectException nsoe) {
592                 Syslog.warn(nsoe);
593                 if (nsoe.detail != null) {
594                     Syslog.warn(nsoe.detail);
595                 }
596                 else {
597                     Syslog.warn("No detail available");
598                 }
599             }
600 
601             Syslog.info(cluster.getClusterName() + " bound on " +
602                                    cluster.getServerName() + ':' + port);
603         }
604         else {
605             throw new IOException("Failed to connect to a valid RMI registry");
606         }
607         return reg;
608     }

624     protected void setUpMulticast(InetAddress group,int port, 
625                         String clusterName,String host,String netInterface)
626         throws IOException {
627         
628         Syslog.info("Setting up Multicast");
629     
630         if (mSock == null) {
631             mSock = new MulticastSocket(port);
632             
633             if (host == null) {
634                 host = getHostName();
635             }
636             else {
637                 mHostName = host;
638             }       
639 
640             //see which interface we're using.
641             InetAddress interf = mSock.getInterface();
642                 
643             Syslog.debug("current interface IP: "

645

647

649                     InetAddress address = LocalNetResolver
650                         .resolveLocalNet(host, netInterface);
651            
652                     if (address != null) {
653               
654                         mSock.setInterface(address);
655                         //change the host to make sure it's on the bc
656                         host = address.getHostAddress();
657 
658                         Syslog.info("new interface IP: " + host);
659 
660                         mHostName = host;
661                     }
662                 }
663                 catch (IOException ioe) {
664                     Syslog.warn(ioe);
665                 }
666             }
667 
668             mSock.setSendBufferSize(1024);
669             mSock.setReceiveBufferSize(1024);
670             mSock.setTimeToLive(1);
671             mSock.joinGroup(group);

676      * converts an array of four bytes to a long. 
677      * @return the converted bytes as a lon or -1 on error.

679     public long convertIPBytes(byte[] ipBytes) {
680         if (ipBytes.length == 4) {
681             long ipLong = (((long)ipBytes[0]) << 24);
682             ipLong |= (((long)ipBytes[1]) << 16);
683             ipLong |= (((long)ipBytes[2]) << 8);
684             ipLong |= (long)ipBytes[3];
685             return ipLong;
686         }
687         return -1;
688     }

690     /**
691      * turns a dot delimited IP address string into a long.
692      */
693     public long convertIPString(String ip) throws NumberFormatException, 
694     IllegalArgumentException {
695 
696         StringTokenizer st = new StringTokenizer(ip,".");
697         if (st.countTokens() == 4) {
698             long ipLong = (Long.parseLong(st.nextToken()) << 24);
699             ipLong += (Long.parseLong(st.nextToken()) << 16);
700             ipLong += (Long.parseLong(st.nextToken()) << 8);
701             ipLong += Long.parseLong(st.nextToken());
702             return ipLong;
703         }
704         else {
705             throw new IllegalArgumentException("Invalid IP string");
706         }
707     }

712     public String convertIPBackToString(long ip) {
713         StringBuffer sb = new StringBuffer(16);
714         sb.append(Long.toString((ip >> 24)& 0xFF));
715         sb.append('.');
716         sb.append(Long.toString((ip >> 16)& 0xFF));
717         sb.append('.');
718         sb.append(Long.toString((ip >> 8) & 0xFF));
719         sb.append('.');
720         sb.append(Long.toString(ip & 0xFF));
721         return sb.toString();
