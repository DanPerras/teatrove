88      public void startEngine(PropertyMap properties,
89                              ServletContext servletContext,
90                              String servletName,
91                              Log log,
92                              List memLog,
93                              PluginContext plug)
94          throws ServletException {
95  
96          try {
97              setProperties(properties);
98              setServletContext(servletContext);
99              setServletName(servletName);
100             setLog(log);
101             setLogEvents(memLog);
102             setPluginContext(plug);
103             mApplicationDepot = new ApplicationDepot(this);
104             
105             // Initialize the HttpContext JMX angent
106             if (properties.getBoolean("management.httpcontext", false) != false) {
107                 MBeanServer mBeanServer = ManagementFactory.getPlatformMBeanServer();
108                 ObjectName objectName = new ObjectName("org.teatrove.teaservlet:name=HttpContext");
109                 if (mBeanServer.isRegistered(objectName) != false) {
110                     mLog.debug("MBean already registered for HttpContext");
111                 }
112                 else {
113                     int cacheSize = properties.getInt("management.httpcontext.readUrlCacheSize", 500);
114                     mBeanServer.registerMBean((HttpContextManagementMBean)HttpContextManagement.create(cacheSize), 
115                                               objectName);
116                 }
117             }
118             
119             // compile templates
120             compileTemplates();
121         }
122         catch (Exception e) {
123             throw new ServletException(e);
124         }
125     }

198     public TeaServletTransaction createTransaction
199         (HttpServletRequest request, HttpServletResponse response,
200          boolean lookupTemplate)
201         throws IOException {
202 
203 
204         try {
205 
206             TeaServletTemplateSource templateSrc = getTemplateSource();
207 
208             TeaServletContextSource contextSrc = (TeaServletContextSource)
209                 templateSrc.getContextSource();
210 
211             Template template = null;
212 
213             if (lookupTemplate) {
214                 // get template path
215                 String path;
216                 if ((path = request.getPathInfo()) == null) {
217                     if ((path = request.getServletPath()) == null) {
218                         path = "/";
219                     }
220                     else {
221                         // Strip off any extension.
222                         int index = path.lastIndexOf('.');
223                         if (index >= 0) {
224                             path = path.substring(0, index);
225                         }
226                     }
227                 }
228 
229                 if (DEBUG) {
230                     // add some comic relief!
231                     mLog.debug("aagggghhh... i've been hit! (" + path + ")");
232                     mLog.debug("Finding template for " + path);
233                 }
234 
235                 // Find the matching template.
236                 template = findTemplate(path, request, response, templateSrc);
237             }
238 
239             // Wrap the user's http response.
240             ApplicationResponse appResponse =
241                 new ApplicationResponseImpl(response, this);
242 
243 
244             ApplicationRequest appRequest =
245                 (lookupTemplate
246                  ? (new ApplicationRequestImpl
247                      (request,
248                       contextSrc
249                       .getApplicationContextTypes(),
250                       template))
251                  : (new ApplicationRequestImpl
252                      (request,
253                       contextSrc.getApplicationContextTypes(),
254                       templateSrc.getTemplateLoader())));
255 
256             try {
257                 /*
258                  * TODO: I dislike this circular logic of having the context
259                  * contain the response and the response containing the
260                  * context, changing this will require a redesign of the
261                  * response.
262                  */
263                 ((ApplicationResponseImpl)appResponse)
264                     .setRequestAndHttpContext(createHttpContext(appRequest,
265                                                                 appResponse,
266                                                                 contextSrc),
267                                               appRequest);
268             }
269             catch (Exception e) {
270                 // Initialize the JDK 1.4+ exception chain,
271                 //   otherwise ServletException for Servlet 2.3 swallows the cause
272                 //noinspection ThrowableInstanceNeverThrown
273                 throw (ServletException) new ServletException(e);
274             }
275 
276             return new RequestAndResponse(appRequest, appResponse);
277 
278         }
279         catch (ServletException se) {
280             // If we are going to swallow the exception, at least log it.
281             getLog().warn(se);
282             return null;
283         }
284     }

301     /**
302      * This method performs the steps necessary to load a new/updated
303      * context source.  It does not swap the mTemplateSource member
304      * until it is fully initialized (its templates have been
305      * compiled).
306      */
307     public TemplateCompilationResults reloadContextAndTemplates(boolean all)
308         throws Exception {
309 
310         // Create the new merged context source
311         TeaServletContextSource contextSource =
312             new TeaServletContextSource(
313                 mApplicationDepot.getClass().getClassLoader(),
314                 mApplicationDepot,
315                 getServletContext(),
316                 getLog(),
317                 true,
318                 mProperties.getBoolean("management.httpcontext", false),
319                 mProperties.getInt("management.httpcontext.readUrlCacheSize", 500),
320                 mProperties.getBoolean("profiling.enabled", true));
321 
322         // Create a new template source using the newly loaded context
323         // source from the new application depot
324         TeaServletTemplateSource templateSrc =
325             createTemplateSource(contextSource);
326 
327         // Recompile all templates using the new context source
328         TemplateCompilationResults results =
329             templateSrc.compileTemplates(null, all);
330 
331         // The new context source and template source are ready for
332         // use, set the member variables
333         synchronized(this) {
334             mApplicationDepot.setContextSource(contextSource);
335             mTemplateSource = templateSrc;
336         }
337 
338         // Return the compilation results
339         return results;
340     }

403     public Template findTemplate(String uri,
404                                  HttpServletRequest request,
405                                  HttpServletResponse response,
406                                  TemplateSource templateSrc)
407         throws ServletException, IOException {
408 
409         Template template = null;
410         try {
411             // If path ends in a slash, revert to loading default template. If
412             // default not found or not specified, return null.
413             boolean useDefault = uri.endsWith("/");
414 
415             // Trim slashes and replace with dots.
416             while (uri.startsWith("/")) {
417                 uri = uri.substring(1);
418             }
419             while (uri.endsWith("/")) {
420                 uri = uri.substring(0, uri.length() - 1);
421             }
422             String name = uri.replace('/', '.');
423 
424             // Look up template if not trying to use default.
425             if (!useDefault) {
426                 // Find template that matches the uri
427                 try {
428                     template = templateSrc.getTemplate(name);
429                 }
430                 catch (ClassNotFoundException e) {
431                     mLog.debug("Can't find template \"" + name + "\": " + e);
432                     template = null;
433                 }
434             }
435 
436             // Use default if no template found so far.
437             if ((template == null)
438                 && (templateSrc instanceof TeaServletTemplateSource)) {
439                 TeaServletTemplateSource tsTsrc = (TeaServletTemplateSource)
440                     templateSrc;
441                 if (tsTsrc.getDefaultTemplateName() != null) {
442                     if  (name.length() == 0) {
443                         name = tsTsrc.getDefaultTemplateName();
444                     }
445                     else {
446                         name = name + '.'
447                             + tsTsrc.getDefaultTemplateName();
448                     }
449                 }
450                 try {
451                     template = tsTsrc.getTemplate(name);
452 
453                     // Redirect if no slash on end of URI.
454                     if (template != null && !useDefault) {
455                         StringBuffer location =
456                             new StringBuffer(request.getRequestURI());
457                         int length = location.length();
458                         if (length == 0
459                             || location.charAt(length - 1) != '/') {
460                             location.append('/');
461                         }
462                         String query = request.getQueryString();
463                         if (query != null) {
464                             location.append('?').append(query);
465                         }
466                         response.setStatus(response.SC_MOVED_PERMANENTLY);
467                         response.sendRedirect(location.toString());
468                     }
469                 }
470                 catch (ClassNotFoundException e) {
471                     mLog.debug("Can't find default template \"" +
472                                name + "\": " + e);
473                 }
474             }
475         }
476         catch (NoClassDefFoundError e) {
477             // The file system let the class load, but because some file
478             // systems support multiple ways in which a file can be found, the
479             // class's exact name may not match. Just report the template
480             // as not being found.
481             return null;
482         }
483         catch (NoSuchMethodException e) {
484             // Initialize the JDK 1.4+ exception chain,
485             //   otherwise ServletException for Servlet 2.3 swallows the cause
486             //noinspection ThrowableInstanceNeverThrown
487             throw (ServletException) new ServletException("Template at \"" + uri
488                                        + "\" is invalid", e);
489         }
490         catch (LinkageError e) {
491             // Initialize the JDK 1.4+ exception chain,
492             //   otherwise ServletException for Servlet 2.3 swallows the cause
493             //noinspection ThrowableInstanceNeverThrown
494             throw (ServletException) new ServletException("Template at \"" + uri
495                                        + "\" is invalid", e);
496         }
497         return template;
498     }
