17  package org.teatrove.teaservlet;
18  
19  import java.util.*;
20  import javax.servlet.*;
21  import org.teatrove.trove.log.Log;
22  import org.teatrove.trove.util.PropertyMap;
23  import org.teatrove.trove.util.plugin.Plugin;
24  
25  /**
26   * An ApplicationConfig implementation that is used for deriving configuration
27   * for internal Applications.
28   *
29   * @author Brian S O'Neill
30   */
31  class InternalApplicationConfig implements ApplicationConfig {  
32      private ApplicationConfig mBaseConfig;
33      private PropertyMap mProperties;
34      private Log mLog;
35      private Map mPluginMap;
36  
37      /**
38       * Derive a new ApplicationConfig using the configuration passed in. Any
39       * properties in the "log" sub-map are applied to the new Log, and the
40       * "init" sub-map defines the new properties. Properties not contained
41       * in "applications" from the base configuration are added as defaults to
42       * to the new properties.
43       *
44       * @param config base configuration to derive from
45       * @param properties properties to use
46       * @param name name of internal application; is applied to Log name and
47       * is used to extract the correct properties.
48       */
49      public InternalApplicationConfig(ApplicationConfig config,
50                                       PropertyMap properties,
51                                       Map plugins,
52                                       String name) {
53          mBaseConfig = config;
54          mLog = new Log(name, config.getLog());
55          mLog.applyProperties(properties.subMap("log"));
56          mProperties = new PropertyMap(properties.subMap("init"));
57  
58          PropertyMap baseProps = config.getProperties();
59          String filtered = "applications" + baseProps.getSeparator();
60          Iterator it = baseProps.keySet().iterator();
61          while (it.hasNext()) {
62              String key = (String)it.next();
63              if (!key.startsWith(filtered) && !mProperties.containsKey(key)) {
64                  mProperties.put(key, baseProps.get(key));
65              }
66          }
67  
68          mPluginMap = plugins;
69      }
70  
71      /**
72       * Returns initialization parameters in an easier to use map format.
73       */
74      public PropertyMap getProperties() {
75          return mProperties;
76      }
77  
78      /**
79       * Returns the name of this application, which is the same as the log's
80       * name.
81       */
82      public String getName() {
83          return mLog.getName();
84      }
85  
86      /**
87       * Returns a log object that this application should use for reporting
88       * information pertaining to the operation of the application.
89       */
90      public Log getLog() {
91          return mLog;
92      }
93      
94      public Plugin getPlugin(String name) {
95          return (Plugin)mPluginMap.get(name);
96      }
97      
98      public Map getPlugins() {       
99          return mPluginMap;
100     }
101 
102     public ServletContext getServletContext() {
103         return mBaseConfig.getServletContext();
104     }
105     
106     public String getInitParameter(String name) {
107         return mProperties.getString(name);
108     }
109     
110     public Enumeration getInitParameterNames() {
111         return Collections.enumeration(mProperties.keySet());
112     }
113 
114     public String getServletName() {
115         return mBaseConfig.getServletName();
116     }
117 }
