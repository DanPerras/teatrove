17  package org.teatrove.teaservlet;
18  
19  import javax.servlet.ServletContext;
20  
21  import org.teatrove.tea.runtime.OutputReceiver;
22  import org.teatrove.tea.engine.DynamicContextSource;
23  import org.teatrove.teaservlet.management.HttpContextManagement;
24  import org.teatrove.trove.io.CharToByteBuffer;
25  import org.teatrove.trove.log.Log;
26  
27  /**
28   * 
29   * @author Jonathan Colwell
30   */
31  public class HttpContextSource implements DynamicContextSource {
32  
33      private ServletContext mServletContext;
34      private Log mLog;
35      private HttpContextManagement mHttpContextMBean;    
36  
37      HttpContextSource(ServletContext servletContext, Log log, HttpContextManagement contextMBean) {
38          mLog = log;
39          mServletContext = servletContext;
40          mHttpContextMBean = contextMBean;
41      }
42  
43      public Class getContextType() {
44          return HttpContext.class;
45      }
46  
47      public Object createContext(Class expected, Object obj) 
48          throws ClassNotFoundException {
49          
50          if (expected.isAssignableFrom(HttpContextImpl.class)) {
51              return createContext(obj);
52          }
53          else {
54              throw new ClassNotFoundException
55                  (expected + " is not available from this context source.");
56          }
57      }
58  
59      public Object createContext(Object obj) {
60  
61          ApplicationRequest req = null;
62          ApplicationResponse resp = null;
63          CharToByteBuffer buf = null;
64          OutputReceiver outRec = null;
65  
66          if (obj instanceof TeaServletTransaction) {
67              TeaServletTransaction trans = (TeaServletTransaction)obj;
68              req = trans.getRequest();
69              resp = trans.getResponse();
70              buf = resp.getResponseBuffer();
71              outRec = trans.getOutputReceiver();
72          }
73          else if (obj instanceof OutputReceiver) {
74              outRec = (OutputReceiver)obj;
75          }
76          return new HttpContextImpl(mServletContext, mLog, req, resp, buf, outRec, mHttpContextMBean);
77      }
78  }
