17  package org.teatrove.teaservlet;

19  import java.io.Serializable;
20  import javax.servlet.http.*;
21  import org.teatrove.teaservlet.io.CharToByteBuffer;
22  import org.teatrove.tea.runtime.OutputReceiver;
23  import org.teatrove.tea.runtime.Substitution;

25  /**
26   * An ordinary HttpServletResponse, but with additional operations specific
27   * to the TeaServlet.
28   *
29   * @author Brian S O'Neill
30   */
31  public interface ApplicationResponse extends HttpServletResponse {
32      /**
33       * Returns whether a redirect or error will be sent to the client. This
34       * is set to true when sendError or sendRedirect is called on the response.
35       */
36      public boolean isRedirectOrError();
37  
38      /**
39       * Returns the internal buffer that stores the response. Applications can
40       * use this to directly supply character or byte data to be output.
41       */
42      public CharToByteBuffer getResponseBuffer();
43  
44      /**
45       * Provides direct access to the HttpContext that the template has access
46       * to. This allows functions to directly control locale and formatting
47       * settings as well as perform string conversion against the current
48       * format settings.
49       * <p>
50       * The returned HttpContext instance is actually an auto-generated class
51       * that merges all the application contexts into one. Application context
52       * instances are requested only when the merged context first needs to
53       * invoke a function from that context.
54       * <p>
55       * Application functions can access functions provided by other
56       * applications, possibly using the Java reflection APIs. Also, the merged
57       * class implements as many context interfaces as possible. Therefore,
58       * casting the context to an expected type can also be used to access other
59       * application functions.
60       */
61      public HttpContext getHttpContext();
62  
63      /**
64       * Execute the given template substitution, but steal any output that would
65       * have gone directly to the response buffer. Applications can use this
66       * to specially handle the output of special objects. For example, an
67       * application can supply functions that allow templates to create
68       * downloadable images. By stealing the output, printed text can go to the
69       * image instead of corrupting the encoding.
70       * <p>
71       * Note: stealOutput is designed to run in the same thread that executed
72       * the main template. If the substitution needs to run in a separate
73       * thread, consider {@link #execDetached} instead.
74       *
75       * @param s template substitution block that will be executed
76       * @param receiver receives all the output generated by the substitution
77       * @throws any exception thrown by the substitution or receiver
78       */
79      public void stealOutput(Substitution s, OutputReceiver receiver)
80          throws Exception;
81  
82      /**
83       * Execute the given template substitution and detach everything sent to
84       * the ApplicationResponse in a way that's safe to do asynchronously.
85       * This is useful for certain caching strategies. The substitution is
86       * detached and executed with a new context instance and output buffer.
87       *
88       * @param s template substitution block that will be executed
89       * @return the detached response data, which is essentially a copy
90       * @throws any exception thrown by the substitution or buffer
91       */
92      public DetachedData execDetached(Substitution s) throws Exception;
93  
94      /**
95       * Execute the given command and detach everything sent to the
96       * ApplicationResponse in a way that's safe to do asynchronously.
97       * This is useful for certain caching strategies. The ApplicationResponse
98       * given to the command contains a new context instance and output buffer.
99       *
100      * @param s template substitution block that will be executed
101      * @return the detached response data, which is essentially a copy
102      * @throws any exception thrown by the command or buffer
103      */
104     public DetachedData execDetached(Command command) throws Exception;
105 
106     /**
107      * Insert an arbitrary command into this response. If this response is
108      * saving its state in a DetachedData instance, then the command
109      * will be executed every time the DetachedData is played back.
110      * If this response isn't detached, the command is not executed and false
111      * is returned.
112      *
113      * @return false if command cannot be inserted because no detached
114      * execution is in progress.
115      */
116     public boolean insertCommand(Command command) throws Exception;
117 
118     /**
119      * Commits this response and fully writes out the buffered contents unless
120      * there was a redirect or an error. Subsequent calls to finish have no
121      * affect.
122      *
123      * @throws IllegalArgumentException if a Writer was already used for
124      * writing to the response.
125      */
126     public void finish() throws java.io.IOException;
127 
128     public interface DetachedData extends Serializable {
129         public void playback(ApplicationRequest request,
130                              ApplicationResponse response) throws Exception;
131         
132         /**
133          * Compresses all the byte buffers using a deflater, unless already
134          * called before. When this DetachedData is later played back for a
135          * request that accepts a compressed encoding, the compressed data is
136          * output. If the request doesn't accept compression, the original
137          * uncompressed data is output.
138          * <p>
139          * The deflater will usually use default strategy, default compression.
140          * If the amount of data to compress is small, the "no compression" (0)
141          * level is used.
142          */
143         public void compress();
144 
145         /**
146          * Compresses all the byte buffers using a deflater, unless already
147          * called before. When this DetachedData is later played back for a
148          * request that accepts a compressed encoding, the compressed data is
149          * output. If the request doesn't accept compression, the original
150          * uncompressed data is output.
151          * <p>
152          * If the amount of data to compress is small, the "no compression" (0)
153          * level is used.
154          *
155          * @param level compression level 1 to 9. 1 is fastest, 9 offers best
156          * compression. 6 is default.
157          */
158         public void compress(int level);
159     }
160 
161     public interface Command {
162         public void execute(ApplicationRequest request,
163                             ApplicationResponse response) throws Exception;
164     }
