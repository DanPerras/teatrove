17  package org.teatrove.teaservlet;
18  
19  import org.teatrove.teatools.*;
20  
21  import java.beans.*;
22  
23  /**
24   * A Tea Tool's best friend.  This class has several useful methods for writing
25   * tools that work with Tea.  Many of these methods were taken from Kettle 
26   * 3.0.x so that they could be reused in future versions and in other 
27   * applications.
28   * <p>
29   * This class was written with the intent that it could be used as a tea
30   * context class. It provides a collection of functions to make introspection
31   * possible from Tea.  
32   *
33   * @author Mark Masse
34   */
35  public interface TeaToolsContext {
36  
37      /**
38       * Returns a bean full of handy information about the specified class.
39       */
40      public HandyClassInfo getHandyClassInfo(Class clazz);
41  
42      /**
43       * Returns a bean full of handy information about the specified class.
44       */
45      public HandyClassInfo getHandyClassInfo(String className);
46  
47      /**
48       * Returns the first sentence of the specified paragraph.  Uses
49       * <code>java.text.BreakIterator.getSentenceInstance()</code>
50       */
51      public String getFirstSentence(String paragraph);
52  
53      /**
54       * Creates a String with the specified pattern repeated length
55       * times.
56       */
57      public String createPatternString(String pattern, int length);
58  
59      /**
60       * Creates a String of spaces with the specified length.
61       */
62      public String createWhitespaceString(int length);
63  
64      /**
65       * provides a bean to contain an assortment of methods to handle class 
66       * names and properties.
67       */
68      interface HandyClassInfo {    
69      
70           /**
71            * Returns the class name of the specified class.  This method 
72            * provides special formatting for array and inner classes.
73            */
74           public String getFullName();
75          
76          
77          /**
78           * Returns the class name of the specified class.  The class name returned
79           * does not include the package. This method provides special formatting 
80           * for array and inner classes.
81           */
82          public String getName();
83  
84          /**
85           * Returns the package name of the specified class.  Returns null if the
86           * class has no package.
87           */
88          public String getPackage();
89  
90          /**
91           * Returns the type.
92           */
93          public Class getType();
94          
95          /**
96           * Returns the name of the type of the Class described by this 
97           * TypeDescription.
98           * <p>
99           * <UL>
100          * <LI>A Class returns "class"
101          * <LI>An Interface returns "interface"
102          * <LI>An array returns null
103          * <LI>A primitive returns null
104          * </UL>
105          */
106         public String getTypeName();
107    
108         /**
109          * Returns the array type.  Returns the specified class if it is 
110          * not an array.  
111          */
112         public TypeDescription getArrayType();
113 
114         /**
115          * Returns the array dimensions.  
116          * Returns 0 if the specified class is not an array.  
117          */
118         public int getArrayDimensions();
119 
120         /**
121          * Returns the array dimensions String (i.e. "[][][]").  
122          * Returns "" (empty string) if the specified class is not an array.
123          */
124         public String getArrayDimensionsString();
125 
126         /**
127          * Returns the shortDescription or "" if the
128          * shortDescription is the same as the displayName.
129          */
130         public String getDescription();
131         
132         /**
133          * Returns the first sentence of the 
134          * shortDescription.  Returns "" if the shortDescription is the same as
135          * the displayName (the default for reflection-generated 
136          * FeatureDescriptors).  
137          */
138         public String getDescriptionFirstSentence();
139         
140         /**
141          * Create a version information string based on what the build process
142          * provided.  The string is of the form "M.m.r" or 
143          * "M.m.r.bbbb" (i.e. 1.1.0.0004) if the build number can be retrieved.
144          * Returns <code>null</code> if the version string cannot be retrieved.
145          */
146         public String getVersion();
147 
148         /**
149          * Introspects a Java bean to learn about all its properties, 
150          * exposed methods, and events.
151          *
152          * @param beanClass the bean class to be analyzed
153          */
154         public BeanInfo getBeanInfo();
155         
156         /**
157          * Returns the type's MethodDescriptions.
158          */
159         public MethodDescription[] getMethodDescriptions();
160         
161         /**
162          * Returns the type's PropertyDescriptions.
163          */
164         public PropertyDescription[] getPropertyDescriptions();
165         
166         /**
167          * A function that returns an array of all the available properties on
168          * a given class.
169          * <p>
170          * <b>NOTE:</b> If possible, the results of this method should be cached
171          * by the caller.
172          *
173          * @param beanClass the bean class to introspect
174          *
175          * @return an array of all the available properties on the specified class.
176          */
177         public PropertyDescriptor[] getTeaBeanPropertyDescriptors();
178                                                         
179         /**
180          * Gets the MethodDescriptors of the specified context class including
181          * all of the MethodDescriptors for methods declared in the class's 
182          * superclass and interfaces
183          *
184          * @param contextClass the Tea context Class to introspect (any class will
185          * work fine)
186          */
187         public MethodDescriptor[] getTeaContextMethodDescriptors();
188 
189     }
190 }
