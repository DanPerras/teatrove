24  /**
25   * This class keeps track of time series performance data for templates.
26   * 
27   * All data is stored in memory.
28   * 
29   * There are two different types of data:
30   *   * raw data
31   *   * aggregate data
32   *   
33   * The RawData keeps track of the most recent start and end times for
34   * a template invocation in addition to their content lengths. This
35   * data is kept in a circular queue of a specified rawWindowSize. Once
36   * the rawWindowSize capacity is reached it creates and AggregateInterval
37   * which contains a smaller set of descriptive statistics for this template.
38   * 
39   * In a similar fashion the aggregate intervals are stored in a circular
40   * queue.
41   * 
42   * In both cases for raw data and the aggregate intervals when the window
43   * size is reached the newest data overwrites the oldest data.
44   * 
45   * @author Scott Jappinen
46   */

70  	public TemplateStats(String fullTemplateName, int rawWindowSize, int aggregateWindowSize) {
71  		assert(fullTemplateName != null);
72  		assert(rawWindowSize > 0);
73  		assert(aggregateWindowSize > 0);
74  		
75  		this.fullTemplateName = fullTemplateName;
76  		int lastPathIndex = fullTemplateName.lastIndexOf('.');
77  		if (lastPathIndex >= 0) {
78  			this.templateName = fullTemplateName.substring(lastPathIndex + 1);
79  			this.templatePath = fullTemplateName.substring(0, lastPathIndex);
80  		} else {
81  			this.templateName = fullTemplateName;
82  			this.templatePath = "";
83  		}
84  		this.mRawWindowSize = rawWindowSize;
85  		mRawData = new RawData[rawWindowSize];
86  		for (int i=0; i < mRawData.length; i++) 
87  			mRawData[i] = new RawData();
88  		
89  		this.mAggregateWindowSize = aggregateWindowSize;
90  		mAggregateIntervals = new AggregateInterval[aggregateWindowSize];
91  		for (int i=0; i < mAggregateIntervals.length; i++) 
92  			mAggregateIntervals[i] = new AggregateInterval();
93  		
94  		mMilestones = new ArrayList<Milestone>();
95  	}

97  	/** Log a template request.
98  	 * 
99  	 * @param startTime
100 	 * @param stopTime
101 	 * @param contentLength
102 	 * @param params
103 	 */
104 	public synchronized void log(long startTime, long stopTime, long contentLength, Object[] params) {
105 		long elapsedTime = (stopTime - startTime);
106 		mRawData[mCurrentRawIndex].set(startTime, stopTime, contentLength);
107 		mCumulativeServiceTime += elapsedTime;
108         if (elapsedTime > mPeakServiceDuration) {
109         	mPeakServiceDuration = elapsedTime;
110         }
111         mServicedCount++;
112 		if (mCurrentRawIndex == mRawWindowSize - 1) {
113 			long aggregateStartTime = -1;
114 			if (mCurrentAggregateIndex == 0) {
115 				if (mAggregateIntervals[mAggregateIntervals.length - 1].getStartTime() != -1) {
116 					aggregateStartTime = mAggregateIntervals[mAggregateIntervals.length - 1].getEndTime() + 1;
117 				}
118 			} else {
119 				aggregateStartTime = mAggregateIntervals[mCurrentAggregateIndex - 1].getEndTime() + 1;
120 			}
121 			mAggregateIntervals[mCurrentAggregateIndex].compute(mRawData, aggregateStartTime, -1);
122 			if (mCurrentAggregateIndex == mAggregateWindowSize - 1) {
123 				mCurrentAggregateIndex = 0;
124 			} else {
125 				mCurrentAggregateIndex++;
126 			}
127 			mCurrentRawIndex = 0;
128 		} else {
129 			mCurrentRawIndex++;
130 		}
131 	}

240 	 * Returns a copy of the raw data. Some of this data may have
241 	 * already been aggregated. The oldest entries are first in the
242 	 * array and the newest entries are last.
243 	 * 
244 	 * @return the raw data
245 	 */
246 	public synchronized RawData[] getRawData() {
247 		//return rawData;
248 		RawData[] result = null;
249 		// put things back in order
250 		//System.out.println("current raw index: " + mCurrentRawIndex + ", " + mRawData);
251 		if (mCurrentRawIndex == 0) {
252 			if (mRawData[0].startTime == -1) { // case 1 - empty
253 				result = null;
254 			} else { // case 2 full
255 				result = mRawData.clone();
256 			}
257 		} else if (mCurrentRawIndex > 0) {
258 			// case 2 - not empty - not full
259 			if (mRawData[mCurrentRawIndex].startTime == -1) {
260 				result = new RawData[mCurrentRawIndex];
261 				for (int i=0; i < result.length; i++) {
262 					result[i] = mRawData[i].clone();
263 				}
264 			} else { // case 3 - full & wrapping
265 				result = new RawData[mRawData.length];
266 				int j=0;
267 				for (int i=mCurrentRawIndex; i < mRawData.length; i++) {
268 					result[j++] = mRawData[i].clone();
269 				}
270 				
271 				for (int i=0; i < mCurrentRawIndex; i++) {
272 					result[j++] = mRawData[i].clone();
273 				}
274 			}
275 		}
276 		return result;
277 	}

405 	 * @param startTime the start time to filter on.

407 	 * @return an aggregate interval for the raw data.

409     public AggregateInterval getAggregateIntervalForRawData(long startTime, 
410     														long stopTime)
411     {
412     	AggregateInterval result = null;
413 		RawData[] rawData = getRawData();
414 		if (rawData != null) {
415 			result = new AggregateInterval();
416 			List<RawData> list = new ArrayList<RawData>();
417 			for (int i=0; i < rawData.length; i++) {
418 				if (rawData[i].startTime >= startTime && rawData[i].endTime <= stopTime) {
419 					list.add(rawData[i]);
420 				}
421 			}
422 			if (list.size() > 0) {
423 				result = new AggregateInterval();
424 				rawData = (RawData[]) list.toArray(new RawData[list.size()]);
425 				result.compute(rawData, startTime, stopTime);
426 			}
427 		}
428     	return result;

472 	/**
473 	 * Returns the aggregate intervals for the specified startTime and stopTime.
474 	 * Any intervals that contain these two endpoints lie between them will be
475 	 * included.
476 	 * 
477 	 * @param startTime the start time to filter on.
478 	 * @param stopTime the stop time to filter on.
479 	 * 
480 	 * @return aggregate intervals for the specified interval.
481 	 */
482 	public AggregateInterval[] getAggregateIntervals(long startTime, long stopTime) {
483 		assert(stopTime >= startTime);
484 		AggregateInterval[] result = null;
485 		AggregateInterval[] allAggregateIntervals = getAggregateIntervals();
486 		if (allAggregateIntervals != null) {
487 			// make sure the interval your looking for 
488 			if (startTime > allAggregateIntervals[allAggregateIntervals.length].getStartTime()) return null;
489 			if (stopTime < allAggregateIntervals[0].getStartTime()) return null;
490 			
491 			int startIndex = search(allAggregateIntervals, startTime);
492 			int stopIndex = search(allAggregateIntervals, stopTime);
493 			
494 			int length = 0;
495 			
496 			if (startIndex == -1) { // comes before aggregate starts
497 				if (stopIndex == -1) { // comes after aggregate ends
498 					// return everything
499 					length = allAggregateIntervals.length;
500 				} else {
501 					length = stopIndex + 1;
502 					startIndex = 0;
503 				}
504 			} else if (stopIndex == -1) {  // comes after aggregate ends
505 				length = allAggregateIntervals.length - startIndex;
506 			} else {
507 				assert(stopIndex >= startIndex);
508 				length = (stopIndex + 1) - startIndex;
509 	
510 			}
511 			
512 			if (length != allAggregateIntervals.length) {
513 				result =  new AggregateInterval[length];
514 				System.arraycopy(allAggregateIntervals, startIndex, result, 0, length);
515 			} else {
516 				result = allAggregateIntervals;
517 			}
518 		}
519 		return result;
520 	}

534     /**
535      * Searches for a AggregateInterval which contains the time passed in.
536      *
537      * @param intervals
538      *            Sorted array of AggregateIntervals
539      * @param time
540      *            Key to search for
541      * @param begin
542      *            Start posisiton in the index
543      * @param end
544      *            One past the end position in the index
545      * @return Integer index to key. -1 if not found
546      */
547     public static int search(AggregateInterval[] intervals, long time, int begin, int end) {
548         end--;
549 
550         while (begin <= end) {
551             int mid = (end + begin) >> 1;
552             if (intervals[mid].getEndTime() < time)
553                 begin = mid + 1;
554             else if (intervals[mid].getStartTime() > time)
555                 end = mid - 1;
556             else
557                 return mid;
558         }
559 
560         return -1;
561     }

597     	}

607 			mRawData[i] = new RawData();
608

612

614

616 		mCumulativeServiceTime = 0L;
617 		mServicedCount = 0L;

619 		mCurrentRawIndex = 0;
